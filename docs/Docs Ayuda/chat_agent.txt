@dataclass
class ChatSession:
    user_id: str
    session_id: str = field(default_factory=lambda: str(uuid.uuid4()))
    conversation_history: List[Dict] = field(default_factory=list)
    user_preferences: Dict[str, str] = field(default_factory=lambda: {
        'communication_style': 'professional',
        'technical_level': 'intermediate',
        'preferred_format': 'combined',
        'response_length': 'medium'
    })
    personalization_data: Dict[str, Any] = field(default_factory=lambda: {
        'frequent_queries': [],
        'successful_interactions': 0,
        'feedback_history': [],
        'usage_patterns': {},
        'basic_queries_usage': 0,
        'quick_queries_count': 0
    })
    current_chart_configs: List[Dict] = field(default_factory=list)
    chart_interaction_history: List[Dict] = field(default_factory=list)
    chart_preferences: Dict[str, Any] = field(default_factory=lambda: {
        'preferred_chart_type': 'bar',
        'chart_complexity': 'medium',
        'interactive_mode': True
    })
    created_at: datetime = field(default_factory=datetime.now)
    last_active: datetime = field(default_factory=datetime.now)
    
    def add_interaction(self, user_msg: str, agent_resp: Dict, intent: str, satisfaction_rating: Optional[int] = None):
        interaction = {
            'timestamp': datetime.now().isoformat(),
            'user_message': user_msg,
            'agent_response': agent_resp,
            'intent': intent,
            'session_id': self.session_id,
            'satisfaction_rating': satisfaction_rating,
            'charts_generated': len(agent_resp.get('charts', [])),
            'chart_interaction': 'current_chart_config' in user_msg,
            'basic_queries_used': agent_resp.get('basic_queries_used', False),
            'data_sources': agent_resp.get('data_sources', [])
        }
        
        self.conversation_history.append(interaction)
        self.last_active = datetime.now()
        self._update_personalization_data(user_msg, intent, satisfaction_rating, agent_resp)
        
        if len(self.conversation_history) > 50:
            self.conversation_history = self.conversation_history[-50:]
    
    def _update_personalization_data(self, user_msg: str, intent: str, rating: Optional[int], agent_resp: Dict):
        if intent not in self.personalization_data['frequent_queries']:
            self.personalization_data['frequent_queries'].append(intent)
        
        if rating and rating >= 4:
            self.personalization_data['successful_interactions'] += 1
        
        msg_length = len(user_msg.split())
        if msg_length < 5:
            self.user_preferences['response_length'] = 'short'
        elif msg_length > 15:
            self.user_preferences['response_length'] = 'detailed'
        
        charts = agent_resp.get('charts', [])
        if charts:
            chart_types = [chart.get('type', 'bar') for chart in charts]
            if chart_types:
                from collections import Counter
                most_common = Counter(chart_types).most_common(1)
                if most_common:
                    self.chart_preferences['preferred_chart_type'] = most_common[0][0]
        
        if 'cambia' in user_msg.lower() or 'gráfico' in user_msg.lower():
            self.chart_interaction_history.append({
                'timestamp': datetime.now().isoformat(),
                'interaction_type': 'pivot',
                'message': user_msg,
                'rating': rating
            })
        
        if agent_resp.get('basic_queries_used'):
            self.personalization_data['basic_queries_usage'] += 1
        
        quick_keywords = ['resumen', 'cuántos', 'total', 'lista', 'rápido']
        if any(keyword in user_msg.lower() for keyword in quick_keywords):
            self.personalization_data['quick_queries_count'] += 1
    
    def add_chart_config(self, config: Dict[str, Any]):
        self.current_chart_configs.append({
            'config': config,
            'timestamp': datetime.now().isoformat(),
            'session_id': self.session_id
        })
        
        if len(self.current_chart_configs) > 10:
            self.current_chart_configs = self.current_chart_configs[-10:]
    
    def get_recent_context(self, max_interactions: int = 5) -> List[Dict]:
        return self.conversation_history[-max_interactions:]
    
    def get_personalization_context(self) -> Dict[str, Any]:
        return {
            'user_preferences': self.user_preferences,
            'chart_preferences': self.chart_preferences,
            'personalization_data': self.personalization_data,
            'total_interactions': len(self.conversation_history),
            'chart_interactions': len(self.chart_interaction_history),
            'basic_queries_usage': self.personalization_data.get('basic_queries_usage', 0),
            'quick_queries_count': self.personalization_data.get('quick_queries_count', 0),
            'last_activity': self.last_active.isoformat()
        }

class ChatSessionManager:
    def __init__(self):
        self.sessions: Dict[str, ChatSession] = {}
        self.active_websockets: Dict[str, WebSocket] = {}  # ← AÑADIR ESTA LÍNEA
    
    def get_or_create_session(self, user_id: str) -> ChatSession:
        if user_id not in self.sessions:
            self.sessions[user_id] = ChatSession(user_id=user_id)
            logger.info(f"🆕 Nueva sesión creada para usuario {user_id}")
        self.sessions[user_id].last_active = datetime.now()
        return self.sessions[user_id]
    
    # ← AÑADIR ESTOS MÉTODOS:
    def add_websocket_connection(self, user_id: str, websocket):
        """Añade conexión WebSocket para un usuario"""
        self.active_websockets[user_id] = websocket
        logger.info(f"🔌 WebSocket conectado para usuario {user_id}")
    
    def remove_websocket_connection(self, user_id: str):
        """Remueve conexión WebSocket para un usuario"""
        if user_id in self.active_websockets:
            del self.active_websockets[user_id]
            logger.info(f"🔌 WebSocket desconectado para usuario {user_id}")
    
    def get_websocket_connection(self, user_id: str):
        """Obtiene conexión WebSocket activa para un usuario"""
        return self.active_websockets.get(user_id)
    
    def reset_session(self, user_id: str):
        if user_id in self.sessions:
            self.sessions[user_id].conversation_history.clear()
            self.sessions[user_id].chart_interaction_history.clear()
            self.sessions[user_id].current_chart_configs.clear()

    # ========================================================================
    # MÉTODOS PRINCIPALES DE PROCESAMIENTO
    # ========================================================================
    
    async def process_chat_message(self, chat_message: ChatMessage) -> ChatResponse:
        """Procesamiento principal de mensajes de chat"""
        start_time = datetime.now()
        session = self.session_manager.get_or_create_session(chat_message.user_id)
        
        try:
            # Sanitizar mensaje
            message = self._sanitize_message(chat_message.message)
            if not message:
                return self._create_error_response("Mensaje vacío", session.session_id, start_time)
            
            # Detectar interacciones con gráficos
            if self._is_chart_interaction(message, chat_message.current_chart_config):
                logger.info("📊 Detectada interacción con gráfico")
                return await self._handle_chart_interaction(chat_message, session, start_time)
            
            # Detectar consultas básicas
            if chat_message.quick_mode or (chat_message.use_basic_queries and self._is_basic_query_request(message)):
                logger.info("📈 Detectada consulta básica")
                return await self._handle_basic_query(chat_message, session, start_time)
            
            # Clasificar intención
            intent_analysis = await self._classify_user_intent(message, session)
            intent = intent_analysis['intent']
            confidence = intent_analysis['confidence']
            
            # Verificar si necesita aclaración
            if self._should_request_clarification(intent_analysis):
                logger.info(f"🔍 Solicitando aclaración - Intent: {intent}, Confianza: {confidence:.2f}")
                return await self._request_clarification(chat_message, intent_analysis, session)
            
            # Extraer parámetros
            gestor_id = self._extract_gestor_id(chat_message, session)
            periodo_final = chat_message.periodo or self._extract_periodo_from_message(message)
            
            # Decisión: CDG Agent o simple
            if intent_analysis['requires_cdg_agent']:
                logger.info("🎯 DELEGANDO A CDG AGENT")
                return await self._process_with_cdg_agent_professional(
                    chat_message, message, gestor_id, periodo_final, session, start_time, intent_analysis
                )
            else:
                logger.info("💬 PROCESANDO CON CHAT PROFESIONAL")
                return await self._process_simple_chat_professional(
                    chat_message, message, gestor_id, periodo_final, session, start_time, intent_analysis
                )
            
        except Exception as e:
            logger.error(f"❌ Error procesando chat: {e}")
            return self._create_error_response(str(e), session.session_id, start_time)
    
    # ========================================================================
    # MÉTODOS DE DETECCIÓN Y CLASIFICACIÓN
    # ========================================================================
    
    def _is_basic_query_request(self, message: str) -> bool:
        """Detecta si es una consulta básica"""
        basic_keywords = [
            'resumen', 'summary', 'general', 'overview',
            'cuántos', 'total', 'cantidad', 'count',
            'ranking', 'top', 'mejor', 'peor',
            'centros', 'oficinas', 'gestores',
            'productos', 'contratos', 'clientes',
            'rápido', 'quick', 'fast',
            'lista', 'listar', 'mostrar'
        ]
        
        message_lower = message.lower()
        return any(keyword in message_lower for keyword in basic_keywords)
    
    def _is_chart_interaction(self, message: str, current_chart_config: Optional[Dict] = None) -> bool:
        """Detecta si el mensaje es una interacción con gráfico"""
        chart_keywords = [
            'cambia', 'cambiar', 'modifica', 'modificar', 'convierte', 'convertir',
            'gráfico', 'chart', 'barras', 'línea', 'circular', 'pie', 'líneas',
            'mostrar', 'visualizar', 'ver en', 'como', 'tipo', 'formato'
        ]
        
        message_lower = message.lower()
        has_chart_keywords = any(k in message_lower for k in chart_keywords)
        return has_chart_keywords and (current_chart_config is not None)

    
    
    async def _classify_user_intent(self, message: str, session: ChatSession) -> Dict[str, Any]:
        """Clasificación de intención con prompts profesionales"""
        try:
            # Detectar interacciones específicas primero
            if self._is_chart_interaction(message):
                return {
                    'intent': 'chart_pivot',
                    'confidence': 0.95,
                    'requires_cdg_agent': False,
                    'recommended_approach': 'chart_interaction'
                }
            
            if self._is_basic_query_request(message):
                return {
                    'intent': 'basic_query',
                    'confidence': 0.88,
                    'requires_cdg_agent': False,
                    'recommended_approach': 'basic_queries'
                }
            
            # Clasificación con LLM
            client = self.llm_client
            
            classification_context = {
                'message': message,
                'recent_history': [h.get('intent', 'general') for h in session.get_recent_context(3)],
                'user_preferences': session.user_preferences,
                'chart_preferences': session.chart_preferences,
                'basic_queries_usage': session.personalization_data.get('basic_queries_usage', 0),
                'session_length': len(session.conversation_history)
            }
            
            response = client.chat.completions.create(
                model=MODEL_NAME,
                messages=[
                    {
                        "role": "system", 
                        "content": CHAT_INTENT_CLASSIFICATION_PROMPT
                    },
                    {
                        "role": "user", 
                        "content": f"Clasifica la siguiente consulta:\n\nMensaje: {message}\n\nContexto: {json.dumps(classification_context, ensure_ascii=False, indent=2)}"
                    }
                ],
                temperature=0.1,
                max_tokens=200
            )
            
            result_text = response.choices[0].message.content.strip()
            
            try:
                result = json.loads(result_text)
                intent = result.get('intent', 'general_inquiry')
                confidence = result.get('confidence', 0.5)
            except json.JSONDecodeError:
                intent = 'general_inquiry'
                confidence = 0.3
            
            logger.info(f"🎯 Intención clasificada: {intent} (confianza: {confidence:.2f})")
            
            return {
                'intent': intent,
                'confidence': confidence,
                'requires_cdg_agent': self._intent_requires_cdg_agent(intent, message),
                'recommended_approach': self._get_approach_for_intent(intent)
            }
            
        except Exception as e:
            logger.error(f"❌ Error en clasificación de intención: {e}")
            return {
                'intent': 'general_inquiry',
                'confidence': 0.3,
                'requires_cdg_agent': self._requires_cdg_agent(message),
                'recommended_approach': 'simple'
            }
    
    # ========================================================================
    # MÉTODOS DE PROCESAMIENTO ESPECÍFICO
    # ========================================================================
    
    async def _handle_basic_query(self, chat_message: ChatMessage, session: ChatSession, start_time: datetime) -> ChatResponse:
        """Maneja consultas básicas usando basic_queries"""
        logger.info("📈 Procesando consulta básica")
        
        try:
            if not self.basic_queries_available:
                return await self._create_basic_query_fallback_response(chat_message, session, start_time)
            
            message_lower = chat_message.message.lower()
            
            # Detectar tipo de consulta básica
            if any(word in message_lower for word in ['resumen', 'summary', 'general', 'overview']):
                data = self.basic_queries.get_resumen_general()
                response_type = "quick_summary"
                response_text = self._format_basic_summary(data, session)
                
            elif any(word in message_lower for word in ['ranking', 'top', 'mejor', 'gestores']):
                metric = 'contratos'
                if any(word in message_lower for word in ['clientes', 'clients']):
                    metric = 'clientes'
                
                data = (self.basic_queries.count_contratos_by_gestor() if metric == 'contratos' 
                       else self.basic_queries.count_clientes_by_gestor())
                response_type = "gestores_ranking"
                response_text = self._format_gestores_ranking(data, metric, session)
                
            else:
                # Consulta genérica
                data = self.basic_queries.get_resumen_general()
                response_type = "quick_summary"
                response_text = self._format_basic_summary(data, session)
            
            execution_time = (datetime.now() - start_time).total_seconds()
            
            return ChatResponse(
                response=response_text,
                response_type=response_type,
                charts=[],
                recommendations=self._get_basic_query_recommendations(response_type),
                metadata={
                    'source': 'basic_queries',
                    'basic_queries_available': True,
                    'query_type': response_type,
                    'data_count': len(data) if isinstance(data, list) else len(data.keys()) if isinstance(data, dict) else 1
                },
                execution_time=execution_time,
                confidence_score=0.9,
                timestamp=datetime.now().isoformat(),
                session_id=session.session_id,
                chart_configs=[],

    async def _handle_chart_interaction(self, chat_message: ChatMessage, session: ChatSession, start_time: datetime) -> ChatResponse:

    def _sanitize_message(self, message: str) -> str:
        """Sanitiza mensaje de entrada"""
        if not message or not isinstance(message, str):
            return ""
        
        sanitized = message.strip()
        if len(sanitized) > 2000:
            sanitized = sanitized[:2000] + "..."
            logger.warning("Mensaje truncado por exceder longitud máxima")
        
        return sanitized
    
    def _extract_periodo_from_message(self, message: str) -> Optional[str]:
        """Extracción inteligente de período"""
        periodo_patterns = [
            r'(\d{4}-\d{2})',
            r'(\d{4}/\d{2})',
            r'(\d{2}/\d{4})',
            r'(\d{2}-\d{4})',
            r'periodo\s+(\d{4}-\d{2})',
            r'mes\s+(\d{4}-\d{2})',
            r'en\s+(\d{4}-\d{2})',
            r'del\s+(\d{4}-\d{2})',
        ]
        
        message_lower = message.lower()
        
        for pattern in periodo_patterns:
            match = re.search(pattern, message_lower)
            if match:
                periodo_found = match.group(1)
                logger.info(f"🗓️ Período detectado: '{periodo_found}'")
                
                # Normalizar formato
                if re.match(r'\d{4}-\d{2}', periodo_found):
                    return periodo_found
                elif re.match(r'\d{4}/\d{2}', periodo_found):
                    return periodo_found.replace('/', '-')
                elif re.match(r'\d{2}/\d{4}', periodo_found):
                    parts = periodo_found.split('/')
                    return f"{parts[1]}-{parts[0]}"
                elif re.match(r'\d{2}-\d{4}', periodo_found):
                    parts = periodo_found.split('-')
                    return f"{parts[1]}-{parts[0]}"
        
        # Fallback dinámico
        current_period = datetime.now().strftime('%Y-%m')
        logger.info(f"🗓️ No se detectó período, usando actual: {current_period}")
        return current_period
    
    def _extract_gestor_id(self, chat_message: ChatMessage, session: ChatSession) -> Optional[str]:
        """Extracción inteligente de gestor_id"""
        
        if chat_message.gestor_id:
            return chat_message.gestor_id
        
        if 'gestor_id' in chat_message.context:
            return chat_message.context['gestor_id']
        
        for interaction in reversed(session.get_recent_context(3)):
            if 'gestor_id' in interaction.get('agent_response', {}).get('metadata', {}):
                return interaction['agent_response']['metadata']['gestor_id']
        
        gestor_patterns = [
            r'gestor\s+(\d+)',
            r'id\s+(\d+)', 
            r'#(\d+)',
            r'número\s+(\d+)',
            r'num\s+(\d+)'
        ]
        
        message_lower = chat_message.message.lower()
        
        for pattern in gestor_patterns:
            match = re.search(pattern, message_lower)
            if match:
                potential_id = match.group(1)
                if potential_id.isdigit() and 1 <= int(potential_id) <= 100:
                    logger.info(f"🎯 Gestor ID extraído: {potential_id}")
                    return potential_id
        
        return "1"

            'incentive_analysis',
            'business_review',
            'executive_summary'
        }
        
        if intent in ['chart_pivot', 'basic_query']:
            return False
        
        return intent in cdg_intents or self._requires_cdg_agent(message)
    
    def _requires_cdg_agent(self, message: str) -> bool:
        """Determina si requiere CDG Agent"""
        message_lower = message.lower()
        
        complex_patterns = [
            r'gestor\s+\d+',
            r'cómo\s+está.*gestor',
            r'performance.*gestor',
            r'rendimiento.*gestor',
            r'comparar.*gestores',
            r'business\s+review',
            r'executive\s+summary'
        ]
        
        for pattern in complex_patterns:
            if re.search(pattern, message_lower):
                logger.info(f"🎯 Consulta compleja detectada: {pattern}")
                return True
        
        return False
    
    def _get_approach_for_intent(self, intent: str) -> str:
        """Obtiene enfoque recomendado según intención"""
        approach_map = {
            'performance_analysis': 'cdg_detailed',
            'comparative_analysis': 'cdg_detailed',
            'deviation_detection': 'cdg_alerts',
            'incentive_analysis': 'cdg_calculations',
            'business_review': 'cdg_executive',
            'executive_summary': 'cdg_executive',
            'chart_pivot': 'chart_interaction',
            'basic_query': 'basic_queries',
            'general_inquiry': 'simple_conversational'
        }
        return approach_map.get(intent, 'simple_conversational')
    
    async def _generate_chart_pivot_response(self, pivot_result: Dict, original_message: str, session: ChatSession) -> str:
        """Genera respuesta profesional para pivoteo de gráficos"""
        try:
            if pivot_result.get('status') == 'success':
                new_config = pivot_result.get('new_config', {})
                chart_type = new_config.get('chart_type', 'gráfico')
                
                return f"📊 **Gráfico actualizado:** {original_message}\n\n✅ He cambiado la visualización a **{chart_type}** según tu solicitud."
            else:
                return f"❌ No pude procesar el cambio de gráfico: {pivot_result.get('message', 'Error desconocido')}"
                
        except Exception as e:
            logger.error(f"Error generando respuesta de pivoteo: {e}")
            return f"✅ He procesado tu solicitud de cambio de gráfico: {original_message}"
    
    def _generate_pivot_suggestions_for_chart(self, pivot_result: Dict) -> List[str]:
        """Genera sugerencias de pivoteo específicas para el gráfico actual"""
        try:
            if pivot_result.get('status') != 'success':
                return []
            
            current_type = pivot_result.get('new_config', {}).get('chart_type', 'bar')
            
            suggestions = []
            
            if current_type == 'bar':
                suggestions.extend([
                    "Cambiar a gráfico circular",
                    "Mostrar como líneas de tendencia"
                ])
            elif current_type == 'line':
                suggestions.extend([
                    "Cambiar a barras verticales", 
                    "Mostrar como área sombreada"
                ])
            elif current_type == 'pie':
                suggestions.extend([
                    "Cambiar a barras comparativas",
                    "Mostrar como tabla de datos"
                ])
            
            return suggestions[:3]
            
        except Exception as e:
            logger.warning(f"Error generando sugerencias de pivoteo: {e}")
            return []
    
    async def _generate_professional_response(self, content: Dict[str, Any], original_message: str, cdg_response, session: ChatSession, intent_analysis: Dict) -> str:
        """Genera respuesta profesional combinando datos y explicación"""
        try:
            # Generar respuesta básica
            response_parts = [
                f"📊 **Análisis para:** {original_message}",
                "",
                "🧠 **Interpretación:**",
                "Los datos han sido procesados exitosamente."
            ]
            
            # Añadir información de gráficos si están disponibles
            if hasattr(cdg_response, 'charts') and cdg_response.charts:
                chart_count = len(cdg_response.charts)
                response_parts.append(f"\n📊 **Visualizaciones:** {chart_count} gráfico{'s' if chart_count > 1 else ''} generado{'s' if chart_count > 1 else ''}.")
            
            return "\n".join(response_parts)
            
        except Exception as e:
            logger.error(f"❌ Error generando respuesta profesional: {e}")
            return f"He procesado tu consulta: {original_message}. Los resultados están disponibles."
    
    async def _process_simple_chat_professional(self, chat_message: ChatMessage, message: str, gestor_id: str, periodo: str, session: ChatSession, start_time: datetime, intent_analysis: Dict) -> ChatResponse:
        """Procesamiento simple con enfoque profesional"""
        try:
            response_text = f"💬 He recibido tu consulta: {message}\n\nProcesando con enfoque conversacional profesional."
            
            chat_response = ChatResponse(
                response=response_text,
                response_type=intent_analysis['intent'],
                charts=[],
                recommendations=["Consulta información más específica para análisis detallado"],
                metadata={
                    'source': 'chat_simple_professional',
                    'intent': intent_analysis['intent'],
                    'confidence': intent_analysis['confidence'],
                    'gestor_id': gestor_id,
                    'periodo': periodo
                },
                execution_time=(datetime.now() - start_time).total_seconds(),
                confidence_score=intent_analysis['confidence'],
                timestamp=datetime.now().isoformat(),
                session_id=session.session_id,
                chart_configs=[],
                pivot_suggestions=[],
                basic_queries_used=False,
                data_sources=['simple_queries']
            )
            
            session.add_interaction(message, chat_response.model_dump(), intent_analysis['intent'])
            
            return chat_response
            
        except Exception as e:
            logger.error(f"❌ Error en procesamiento simple profesional: {e}")
            return self._create_error_response(str(e), session.session_id, start_time)
    
    async def _process_user_feedback_professional(self, user_id: str, feedback: Dict[str, Any], session: ChatSession):
        """Procesamiento profesional de feedback"""
        try:
            if not feedback:
                return
            
            rating = feedback.get('rating', 0)
            
            # Añadir a historial
            session.personalization_data['feedback_history'].append({
                'timestamp': datetime.now().isoformat(),
                'rating': rating,
                'feedback': feedback
            })
            
            # Ajustar preferencias basado en rating
            if rating >= 4:
                session.personalization_data['successful_interactions'] += 1
            elif rating <= 2:
                comments = feedback.get('comment', '').lower()
                if 'muy largo' in comments:
                    session.user_preferences['response_length'] = 'short'
                elif 'muy corto' in comments:
                    session.user_preferences['response_length'] = 'detailed'
            
            logger.info(f"📝 Feedback procesado para usuario {user_id}")
            
        except Exception as e:
            logger.error(f"❌ Error procesando feedback: {e}")
