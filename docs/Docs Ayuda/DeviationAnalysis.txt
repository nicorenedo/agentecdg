/**
 * DeviationAnalysis - Análisis completo de desviaciones e incentivos
 * Soporte dual: Dirección (global) y Gestor (personal)
 */
const DeviationAnalysis = ({
  mode = 'direccion',
  periodo,
  gestorId = null,
  bonusPool = 50000,
  filters = {},
  onReload = () => {},
  onSelectRow = () => {},
  onOpenDrillDown = () => {},
  className = '',
  style = {}
}) => {
  // Estados principales
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);
  const [activeTab, setActiveTab] = useState('pricing');
  
  // Estados de datos
  const [summary, setSummary] = useState(null);
  const [deviationPricing, setDeviationPricing] = useState([]);
  const [deviationMargin, setDeviationMargin] = useState([]);
  const [deviationVolume, setDeviationVolume] = useState([]);
  const [incentivesData, setIncentivesData] = useState([]);
  
  // Estados de interacción
  const [selectedEntity, setSelectedEntity] = useState(null);
  const [showOnlyOutliers, setShowOnlyOutliers] = useState(false);

  // Cargar datos principales
  const fetchDeviationData = useCallback(async () => {
    if (!periodo) {
      setError('Período requerido para análisis de desviaciones');
      return;
    }

    setLoading(true);
    setError(null);

    try {
      // 1. Resumen general de desviaciones
      const summaryResponse = await api.deviations.summary(periodo);
      
      // 2. Desviaciones de pricing
      const pricingResponse = await api.deviations.pricing(periodo, 15.0);
      
      // 3. Desviaciones de margen
      const marginResponse = await api.deviations.margen(periodo, { z: 2.0, enhanced: true });
      
      // 4. Desviaciones de volumen
      const volumeResponse = await api.deviations.volumen(periodo, { factor: 3.0, enhanced: true });
      
      // 5. Incentivos según contexto
      let incentivesResponse = [];
      if (mode === 'direccion' && bonusPool) {
        incentivesResponse = await api.incentives.bonusPool(periodo, bonusPool);
      } else if (mode === 'gestor' && gestorId) {
        incentivesResponse = await api.incentives.scorecard(gestorId, periodo);
      }

      // Procesar y normalizar datos
      setSummary(normalizeSummary(summaryResponse));
      setDeviationPricing(normalizePricingData(pricingResponse.deviations || []));
      setDeviationMargin(normalizeMarginData(marginResponse.deviations || []));
      setDeviationVolume(normalizeVolumeData(volumeResponse.deviations || []));
      setIncentivesData(normalizeIncentivesData(incentivesResponse, bonusPool));

    } catch (err) {
      console.error('[DeviationAnalysis] Error loading data:', err);
      setError(err?.message || 'Error al cargar datos de desviaciones');
      
      // Datos mock para demo
      setSummary(generateMockSummary());
      setDeviationPricing(generateMockPricing());
      setDeviationMargin(generateMockMargin());
      setDeviationVolume(generateMockVolume());
      setIncentivesData(generateMockIncentives());
      
    } finally {
      setLoading(false);
    }
  }, [periodo, gestorId, mode, bonusPool]);

  // Normalizadores de datos
  const normalizeSummary = (data) => ({
    totalDeviation: data?.total_deviation || 0,
    deviationPercent: data?.deviation_percent || 0,
    outlierCount: data?.outlier_count || 0,
    criticalAlerts: data?.critical_alerts || 0,
    insight: data?.insight || 'Análisis completado sin incidencias críticas.',
    period: data?.periodo || periodo,
    lastUpdate: new Date().toISOString()
  });

  const normalizePricingData = (data) => {
    if (!Array.isArray(data)) return [];
    return data.map((item, index) => ({
      key: `pricing-${index}`,
      entity: item.producto || item.entity || `Entidad ${index + 1}`,
      target: Number(item.precio_std || item.target || 0),
      actual: Number(item.precio_real || item.actual || 0),
      deviation: Number(item.delta_abs || item.deviation || 0),
      deviationPercent: Number(item.delta_pct || item.deviation_percent || 0),
      semaforo: item.semaforo || (Math.abs(item.delta_pct || 0) > 15 ? 'Rojo' : 
                  Math.abs(item.delta_pct || 0) > 5 ? 'Amarillo' : 'Verde'),
      trend: item.trend || (Math.random() > 0.5 ? 'up' : 'down'),
      contracts: item.contratos || Math.floor(Math.random() * 20) + 1
    }));
  };

  const normalizeMarginData = (data) => {
    if (!Array.isArray(data)) return [];
    return data.map((item, index) => ({
      key: `margin-${index}`,
      entity: item.gestor || item.centro || item.entity || `Entidad ${index + 1}`,
      target: Number(item.margen_objetivo || item.target || 0),
      actual: Number(item.margen_real || item.actual || 0),
      deviation: Number(item.deviation || 0),
      deviationPercent: Number(item.deviation_percent || 0),
      semaforo: item.semaforo || (item.deviation < -10000 ? 'Rojo' : 
                  item.deviation < -5000 ? 'Amarillo' : 'Verde'),
      trend: item.trend || (Math.random() > 0.5 ? 'up' : 'down'),
      clients: item.clientes || Math.floor(Math.random() * 50) + 10
    }));
  };

  const normalizeVolumeData = (data) => {
    if (!Array.isArray(data)) return [];
    return data.map((item, index) => ({
      key: `volume-${index}`,
      entity: item.entity || `Entidad ${index + 1}`,
      target: Number(item.target || 0),
      actual: Number(item.actual || 0),
      deviation: Number(item.deviation || 0),
      deviationPercent: Number(item.deviation_percent || 0),
      semaforo: item.semaforo || (Math.abs(item.deviation_percent || 0) > 20 ? 'Rojo' : 
                  Math.abs(item.deviation_percent || 0) > 10 ? 'Amarillo' : 'Verde'),
      trend: item.trend || (Math.random() > 0.5 ? 'up' : 'down'),
      volume: item.volume || Math.floor(Math.random() * 1000000) + 100000
    }));
  };

  const normalizeIncentivesData = (data, pool) => {
    if (!data || !Array.isArray(data.pool || data.scorecard || [])) return [];
    const items = data.pool || data.scorecard || [];
    const totalContribution = items.reduce((sum, item) => sum + (item.contribution || 0), 0);
    
    return items.map((item, index) => ({
      key: `incentive-${index}`,
      entity: item.gestor || item.entity || `Entidad ${index + 1}`,
      contribution: Number(item.contribution || 0),
      contributionPercent: totalContribution > 0 ? (item.contribution / totalContribution) * 100 : 0,
      estimatedAmount: pool ? (pool * (item.contribution || 0) / totalContribution) : 0,
      performance: item.performance || Math.random() * 100 + 50,
      ranking: index + 1
    }));
  };

  // Datos mock para desarrollo
  const generateMockSummary = () => ({
    totalDeviation: 42500,
    deviationPercent: 8.3,
    outlierCount: 7,
    criticalAlerts: 3,
    insight: 'Se detectan 3 alertas críticas en pricing de productos hipotecarios. Revisar condiciones comerciales del centro Madrid.',
    period: periodo,
    lastUpdate: new Date().toISOString()
  });

  const generateMockPricing = () => [
    { key: 'p1', entity: 'Préstamo Hipotecario', target: -1200, actual: -1185, deviation: 15, deviationPercent: 1.25, semaforo: 'Verde', trend: 'up', contracts: 45 },
    { key: 'p2', entity: 'Depósito Plazo', target: -1160, actual: -1245, deviation: -85, deviationPercent: -7.33, semaforo: 'Rojo', trend: 'down', contracts: 23 },
    { key: 'p3', entity: 'Fondo Banca March', target: -1180, actual: -1165, deviation: 15, deviationPercent: 1.27, semaforo: 'Verde', trend: 'up', contracts: 67 },
    { key: 'p4', entity: 'Cuenta Corriente Plus', target: -950, actual: -1075, deviation: -125, deviationPercent: -13.16, semaforo: 'Rojo', trend: 'down', contracts: 12 }
  ];

  const generateMockMargin = () => [
    { key: 'm1', entity: 'Ana García López', target: 125000, actual: 142000, deviation: 17000, deviationPercent: 13.6, semaforo: 'Verde', trend: 'up', clients: 34 },
    { key: 'm2', entity: 'Carlos Martín', target: 98000, actual: 85500, deviation: -12500, deviationPercent: -12.8, semaforo: 'Rojo', trend: 'down', clients: 28 },
    { key: 'm3', entity: 'María José Ruiz', target: 110000, actual: 118000, deviation: 8000, deviationPercent: 7.3, semaforo: 'Verde', trend: 'up', clients: 31 }
  ];

  const generateMockVolume = () => [
    { key: 'v1', entity: 'Centro Madrid', target: 180, actual: 195, deviation: 15, deviationPercent: 8.3, semaforo: 'Verde', trend: 'up', volume: 2850000 },
    { key: 'v2', entity: 'Centro Barcelona', target: 145, actual: 128, deviation: -17, deviationPercent: -11.7, semaforo: 'Amarillo', trend: 'down', volume: 1950000 }
  ];

  const generateMockIncentives = () => [
    { key: 'i1', entity: 'Ana García López', contribution: 18.5, contributionPercent: 18.5, estimatedAmount: 9250, performance: 142, ranking: 1 },
    { key: 'i2', entity: 'Carlos Martín', contribution: 14.2, contributionPercent: 14.2, estimatedAmount: 7100, performance: 87, ranking: 2 },
    { key: 'i3', entity: 'María José Ruiz', contribution: 12.8, contributionPercent: 12.8, estimatedAmount: 6400, performance: 118, ranking: 3 }
  ];

  // Effect para cargar datos
  useEffect(() => {
    fetchDeviationData();
  }, [fetchDeviationData]);

  // Handlers
  const handleEntitySelect = useCallback((record, type) => {
    setSelectedEntity({ ...record, type });
    onSelectRow({ ...record, type, mode, periodo });
  }, [onSelectRow, mode, periodo]);

  const handleDrillDown = useCallback((record, type) => {
    onOpenDrillDown({ ...record, type, mode, periodo, gestorId });
  }, [onOpenDrillDown, mode, periodo, gestorId]);

  const handleRefresh = useCallback(() => {
    fetchDeviationData();
    onReload();
  }, [fetchDeviationData, onReload]);

  // Filtros para datos
  const getFilteredData = (data) => {
    if (!showOnlyOutliers) return data;
    return data.filter(item => 
      item.semaforo === 'Rojo' || 
      Math.abs(item.deviationPercent || 0) > 10
    );
  };

  // Definición de columnas por tipo
  const pricingColumns = [
    {
      title: mode === 'direccion' ? 'Producto' : 'Cliente/Producto',
      dataIndex: 'entity',
      key: 'entity',
      sorter: (a, b) => (a.entity || '').localeCompare(b.entity || ''),
      render: (text, record) => (
        <Space>
          <Text strong style={{ fontSize: 13 }}>{text}</Text>
          {record.semaforo === 'Rojo' && (
            <AlertOutlined style={{ color: theme.colors.error, fontSize: 12 }} />
          )}
        </Space>
      )
    },
    {
      title: 'Precio Objetivo',
      dataIndex: 'target',
      key: 'target',
      sorter: (a, b) => a.target - b.target,
      render: val => <Text>{val.toFixed(0)}€</Text>
    },
    {
      title: 'Precio Real',
      dataIndex: 'actual',
      key: 'actual',
      sorter: (a, b) => a.actual - b.actual,
      render: val => <Text>{val.toFixed(0)}€</Text>
    },
    {
      title: 'Desviación',
      dataIndex: 'deviationPercent',
      key: 'deviationPercent',
      sorter: (a, b) => Math.abs(a.deviationPercent) - Math.abs(b.deviationPercent),
      render: (val, record) => (
        <Space>
          <Text style={{ 
            color: Math.abs(val) > 15 ? theme.colors.error : 
                   Math.abs(val) > 5 ? theme.colors.warning : theme.colors.success,
            fontWeight: 600
          }}>
            {val > 0 ? '+' : ''}{val.toFixed(1)}%
          </Text>
          <Tag color={record.semaforo === 'Verde' ? 'green' : record.semaforo === 'Amarillo' ? 'orange' : 'red'}>
            {record.semaforo}
          </Tag>
        </Space>
      )
    },
    {
      title: 'Contratos',
      dataIndex: 'contracts',
      key: 'contracts',
      sorter: (a, b) => a.contracts - b.contracts,
      render: val => <Text type="secondary">{val}</Text>
    },
    {
      title: 'Tendencia',
      dataIndex: 'trend',
      key: 'trend',
      render: val => val === 'up' ? 
        <ArrowUpOutlined style={{ color: theme.colors.success }} /> : 
        <ArrowDownOutlined style={{ color: theme.colors.error }} />
    },
    {
      title: 'Acciones',
      key: 'actions',
      render: (_, record) => (
        <Space size="small">
          <Button 
            size="small" 
            onClick={() => handleEntitySelect(record, 'pricing')}
            type={selectedEntity?.key === record.key ? 'primary' : 'default'}
          >
            Seleccionar
          </Button>
          <Button 
            size="small" 
            onClick={() => handleDrillDown(record, 'pricing')}
            icon={<InfoCircleOutlined />}
          >
            Drill-down
          </Button>
        </Space>
      )
    }
  ];

  const marginColumns = [
    {
      title: mode === 'direccion' ? 'Entidad' : 'Cliente',
      dataIndex: 'entity',
      key: 'entity',
      sorter: (a, b) => (a.entity || '').localeCompare(b.entity || ''),
      render: (text, record) => (
        <Space>
          <Text strong style={{ fontSize: 13 }}>{text}</Text>
          {record.semaforo === 'Rojo' && (
            <AlertOutlined style={{ color: theme.colors.error, fontSize: 12 }} />
          )}
        </Space>
      )
    },
    {
      title: 'Objetivo (€)',
      dataIndex: 'target',
      key: 'target',
      sorter: (a, b) => a.target - b.target,
      render: val => <Text>{val.toLocaleString()}€</Text>
    },
    {
      title: 'Real (€)',
      dataIndex: 'actual',
      key: 'actual',
      sorter: (a, b) => a.actual - b.actual,
      render: val => <Text>{val.toLocaleString()}€</Text>
    },
    {
      title: 'Desviación (€)',
      dataIndex: 'deviation',
      key: 'deviation',
      sorter: (a, b) => a.deviation - b.deviation,
      render: (val, record) => (
        <Space>
          <Text style={{ 
            color: val < -5000 ? theme.colors.error : 
                   val < 0 ? theme.colors.warning : theme.colors.success,
            fontWeight: 600
          }}>
            {val > 0 ? '+' : ''}{val.toLocaleString()}€
          </Text>
          <Tag color={record.semaforo === 'Verde' ? 'green' : record.semaforo === 'Amarillo' ? 'orange' : 'red'}>
            {record.semaforo}
          </Tag>
        </Space>
      )
    },
    {
      title: mode === 'direccion' ? 'Contratos' : 'Clientes',
      dataIndex: mode === 'direccion' ? 'contracts' : 'clients',
      key: 'volume',
      sorter: (a, b) => (a.contracts || a.clients) - (b.contracts || b.clients),
      render: val => <Text type="secondary">{val || 0}</Text>
    },
    {
      title: 'Tendencia',
      dataIndex: 'trend',
      key: 'trend',
      render: val => val === 'up' ? 
        <ArrowUpOutlined style={{ color: theme.colors.success }} /> : 
        <ArrowDownOutlined style={{ color: theme.colors.error }} />
    }
  ];

  const volumeColumns = [
    {
      title: mode === 'direccion' ? 'Centro/Segmento' : 'Cliente',
      dataIndex: 'entity',
      key: 'entity',
      sorter: (a, b) => (a.entity || '').localeCompare(b.entity || ''),
      render: (text, record) => (
        <Space>
          <Text strong style={{ fontSize: 13 }}>{text}</Text>
          {record.semaforo === 'Rojo' && (
            <AlertOutlined style={{ color: theme.colors.error, fontSize: 12 }} />
          )}
        </Space>
      )
    },
    {
      title: 'Objetivo',
      dataIndex: 'target',
      key: 'target',
      sorter: (a, b) => a.target - b.target,
    },
    {
      title: 'Real',
      dataIndex: 'actual',
      key: 'actual',
      sorter: (a, b) => a.actual - b.actual,
    },
    {
      title: 'Desviación (%)',
      dataIndex: 'deviationPercent',
      key: 'deviationPercent',
      sorter: (a, b) => Math.abs(a.deviationPercent) - Math.abs(b.deviationPercent),
      render: (val, record) => (
        <Space>
          <Text style={{ 
            color: Math.abs(val) > 20 ? theme.colors.error : 
                   Math.abs(val) > 10 ? theme.colors.warning : theme.colors.success,
            fontWeight: 600
          }}>
            {val > 0 ? '+' : ''}{val.toFixed(1)}%
          </Text>
          <Tag color={record.semaforo === 'Verde' ? 'green' : record.semaforo === 'Amarillo' ? 'orange' : 'red'}>
            {record.semaforo}
          </Tag>
        </Space>
      )
    },
    {
      title: 'Volumen (€)',
      dataIndex: 'volume',
      key: 'volume',
      sorter: (a, b) => (a.volume || 0) - (b.volume || 0),
      render: val => <Text type="secondary">{(val || 0).toLocaleString()}€</Text>
    },
    {
      title: 'Tendencia',
      dataIndex: 'trend',
      key: 'trend',
      render: val => val === 'up' ? 
        <ArrowUpOutlined style={{ color: theme.colors.success }} /> : 
        <ArrowDownOutlined style={{ color: theme.colors.error }} />
    }
  ];

  const incentivesColumns = [
    {
      title: mode === 'direccion' ? 'Gestor' : 'Concepto',
      dataIndex: 'entity',
      key: 'entity',
      sorter: (a, b) => (a.entity || '').localeCompare(b.entity || ''),
      render: (text, record) => (
        <Space>
          <Badge count={record.ranking} style={{ backgroundColor: theme.colors.bmGreenPrimary }}>
            <Text strong style={{ fontSize: 13 }}>{text}</Text>
          </Badge>
        </Space>
      )
    },
    {
      title: 'Contribución (%)',
      dataIndex: 'contributionPercent',
      key: 'contributionPercent',
      sorter: (a, b) => a.contributionPercent - b.contributionPercent,
      render: (val) => (
        <Space>
          <Progress 
            percent={val} 
            size="small" 
            strokeColor={theme.colors.bmGreenPrimary}
            style={{ width: 100 }}
            showInfo={false}
          />
          <Text>{val.toFixed(1)}%</Text>
        </Space>
      )
    },
    {
      title: 'Importe Estimado (€)',
      dataIndex: 'estimatedAmount',
      key: 'estimatedAmount',
      sorter: (a, b) => a.estimatedAmount - b.estimatedAmount,
      render: val => (
        <Text strong style={{ color: theme.colors.bmGreenPrimary }}>
          {val.toLocaleString()}€
        </Text>
      )
    },
    {
      title: 'Performance',
      dataIndex: 'performance',
      key: 'performance',
      sorter: (a, b) => a.performance - b.performance,
      render: val => (
        <Tag color={val > 120 ? 'green' : val > 100 ? 'blue' : val > 80 ? 'orange' : 'red'}>
          {val.toFixed(0)}%
        </Tag>
      )
    }
  ];

  // Renderizado del header de resumen
  const renderSummaryHeader = () => {
    if (!summary) return null;

    return (
      <div style={{ marginBottom: 24 }}>
        <Row gutter={[16, 16]}>
          <Col xs={24} sm={12} md={6}>
            <Card size="small" style={{ textAlign: 'center', borderColor: theme.colors.bmGreenLight }}>
              <Statistic
                title="Desviación Total"
                value={summary.totalDeviation}
                precision={0}
                valueStyle={{ 
                  color: Math.abs(summary.deviationPercent) > 10 ? theme.colors.error : theme.colors.bmGreenPrimary 
                }}
                prefix={<DollarOutlined />}
                suffix="€"
              />
              <Text type="secondary" style={{ fontSize: 12 }}>
                ({summary.deviationPercent > 0 ? '+' : ''}{summary.deviationPercent.toFixed(1)}%)
              </Text>
            </Card>
          </Col>
          
          <Col xs={24} sm={12} md={6}>
            <Card size="small" style={{ textAlign: 'center', borderColor: theme.colors.warning }}>
              <Statistic

                value={summary.outlierCount}
                valueStyle={{ 
                  color: summary.outlierCount > 5 ? theme.colors.error : theme.colors.warning

              />

          <Col xs={24} sm={12} md={6}>
            <Card size="small" style={{ textAlign: 'center', borderColor: theme.colors.error }}>
              <Statistic
                title="Alertas Críticas"
                value={summary.criticalAlerts}
                valueStyle={{ 
                  color: summary.criticalAlerts > 0 ? theme.colors.error : theme.colors.success 
                }}
                prefix={<ExclamationCircleOutlined />}
              />
            </Card>
          </Col>
          
          {bonusPool && mode === 'direccion' && (
            <Col xs={24} sm={12} md={6}>
              <Card size="small" style={{ textAlign: 'center', borderColor: theme.colors.success }}>
                <Statistic
                  title="Pool de Incentivos"
                  value={bonusPool}
                  precision={0}
                  valueStyle={{ color: theme.colors.success }}
                  prefix={<TrophyOutlined />}
                  suffix="€"
                />
              </Card>
            </Col>
          )}
        </Row>
        
        {summary.insight && (
          <Alert
            message="Análisis Automático"
            description={summary.insight}
            type={summary.criticalAlerts > 0 ? 'warning' : 'info'}
            showIcon
            style={{ marginTop: 16 }}
          />
        )}
      </div>
    );

  // ✅ CORRECCIÓN PRINCIPAL: Usar items API en lugar de TabPane
  const tabItems = [
    {
      key: 'pricing',
      label: (
        <Space>
          <DollarOutlined />
          Pricing
          {deviationPricing.filter(item => item.semaforo === 'Rojo').length > 0 && (
            <Badge count={deviationPricing.filter(item => item.semaforo === 'Rojo').length} />

      ),

        <Table
          columns={pricingColumns}
          dataSource={getFilteredData(deviationPricing)}

          scroll={{ x: true }}
          pagination={{
            defaultPageSize: 10,
            pageSizeOptions: ['10', '20', '50'],

      ),
      children: (
        <Table
          columns={marginColumns}
          dataSource={getFilteredData(deviationMargin)}
          size="small"
          scroll={{ x: true }}
          pagination={{
            defaultPageSize: 10,
            pageSizeOptions: ['10', '20', '50'],
            showSizeChanger: true,
            showQuickJumper: true
          }}

      children: (
        <Table
          columns={volumeColumns}
          dataSource={getFilteredData(deviationVolume)}
          size="small"
          scroll={{ x: true }}
          pagination={{
            defaultPageSize: 10,
            pageSizeOptions: ['10', '20', '50'],
            showSizeChanger: true,
            showQuickJumper: true
          }}
        />
      )
    },
    {
      key: 'incentives',
      label: (
        <Space>
          <TrophyOutlined />
          Incentivos
        </Space>
      ),
      children: (
        <Table
          columns={incentivesColumns}
          dataSource={incentivesData}
          size="small"
          scroll={{ x: true }}
          pagination={{
            defaultPageSize: 10,
            pageSizeOptions: ['10', '20', '50'],
            showSizeChanger: true,
            showQuickJumper: true
          }}
        />
      )
    }
  ];

  if (loading) {
    return <Loader />;
  }

  if (error) {
    return (
      <ErrorState 
        error={error} 
        onRetry={handleRefresh}
        style={{ height: '100%' }}
      />
    );
  }

  return (
    <Card
      className={className}
      style={{
        width: '100%',
        height: '100%',
        ...style
      }}
      styles={{ body: { padding: '16px' } }} // ✅ CORRECCIÓN: bodyStyle → styles.body
      title={
        <Space>
          <AlertOutlined style={{ color: theme.colors.bmGreenPrimary }} />
          <Title level={4} style={{ margin: 0 }}>
            Análisis de Desviaciones e Incentivos
          </Title>
          <Badge 
            count={mode === 'direccion' ? 'Dirección' : 'Gestor'} 
            style={{ backgroundColor: theme.colors.bmGreenPrimary }} 
          />
          <Text type="secondary">• {periodo}</Text>
        </Space>
      }
      extra={
        <Space>
          <Switch
            checkedChildren="Solo Outliers"
            unCheckedChildren="Todos"
            checked={showOnlyOutliers}
            onChange={setShowOnlyOutliers}
          />
          <Button 
            icon={<ReloadOutlined />} 
            onClick={handleRefresh}
            disabled={loading}
          >
            Recargar
          </Button>
        </Space>
      }
    >
      {renderSummaryHeader()}
      
      {/* ✅ CORRECCIÓN PRINCIPAL: Usar items en lugar de TabPane */}
      <Tabs
        activeKey={activeTab}
        onChange={setActiveTab}
        tabPosition="top"
        type="line"
        style={{ minHeight: 400 }}
        items={tabItems} // ✅ Nuevo API de Ant Design v5
      />
    </Card>
  );
};

DeviationAnalysis.propTypes = {
  mode: PropTypes.oneOf(['direccion', 'gestor']).isRequired,
  periodo: PropTypes.string.isRequired,
  gestorId: PropTypes.oneOfType([PropTypes.string, PropTypes.number]),
  bonusPool: PropTypes.number,
  filters: PropTypes.object,
  onReload: PropTypes.func,
  onSelectRow: PropTypes.func,
  onOpenDrillDown: PropTypes.func,
  className: PropTypes.string,
  style: PropTypes.object
};

export default DeviationAnalysis;
