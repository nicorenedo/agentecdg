class ResponseFormat(Enum):
    """Formatos de respuesta disponibles"""
    JSON = "json"
    TEXT = "text"
    DASHBOARD = "dashboard"
    BUSINESS_REPORT = "business_report"


@dataclass
class CDGRequest:
    """Estructura de request para el agente CDG"""
    user_message: str
    user_id: Optional[str] = None
    gestor_id: Optional[str] = None
    periodo: Optional[str] = None
    context: Dict[str, Any] = field(default_factory=dict)
    response_format: ResponseFormat = ResponseFormat.JSON
    include_charts: bool = True
    include_recommendations: bool = True
    current_chart_config: Optional[Dict[str, Any]] = None
    chart_interaction_type: Optional[str] = None
    use_basic_queries: bool = True
    quick_mode: bool = False


@dataclass
class CDGResponse:
    """Estructura de respuesta del agente CDG"""
    response_type: QueryType
    content: Dict[str, Any]
    charts: List[Dict[str, Any]] = field(default_factory=list)
    recommendations: List[str] = field(default_factory=list)
    metadata: Dict[str, Any] = field(default_factory=dict)
    execution_time: float = 0.0
    confidence_score: float = 0.0
    created_at: datetime = field(default_factory=datetime.now)
    chart_configs: List[Dict[str, Any]] = field(default_factory=list)
    pivot_suggestions: List[str] = field(default_factory=list)
    basic_queries_used: bool = False
    data_sources: List[str] = field(default_factory=list)

    def to_dict(self) -> Dict[str, Any]:
        """Convierte la respuesta a diccionario"""
        return {
            'response_type': self.response_type.value,
            'content': self.content,
            'charts': self.charts,
            'recommendations': self.recommendations,
            'metadata': self.metadata,
            'execution_time': self.execution_time,
            'confidence_score': self.confidence_score,
            'created_at': self.created_at.isoformat(),
            'chart_configs': self.chart_configs,
            'pivot_suggestions': self.pivot_suggestions,
            'basic_queries_used': self.basic_queries_used,
            'data_sources': self.data_sources
        }


class CDGAgent:
    """
    Agente Principal de Control de Gestión
    
    Orquestador universal integrado con chat_agent que puede manejar:
    - Análisis de gestores y performance
    - Comparativas y rankings
    - Desviaciones y anomalías
    - Incentivos y comisiones
    - Precios estándar vs reales
    - Análisis de centros, segmentos, productos
    - Generación de reportes y dashboards
    - Consultas exploratorias de datos
    - Chat conversacional avanzado
    - Generación y pivoteo de gráficos inteligentes
    """
    
    def __init__(self):
        """Inicializa el agente CDG con todos los módulos"""
        self.start_time = datetime.now()
        
        # Inicializar componentes principales
        self.kpi_calculator = FinancialKPICalculator()
        self.chart_generator = CDGDashboardGenerator()
        self.query_chart_generator = QueryIntegratedChartGenerator()
        self.report_generator = BusinessReportGenerator()
        
        # Inicializar queries
        self.gestor_queries = GestorQueries()
        self.comparative_queries = ComparativeQueries()
        self.deviation_queries = DeviationQueries()
        self.incentive_queries = IncentiveQueries()
        self.period_queries = PeriodQueries()
        self.basic_queries = basic_queries
        
        # Configuración
        self.config_manager = DynamicBusinessConfig()
        self.conversation_history = []
        self.imports_successful = IMPORTS_SUCCESSFUL
        
        # Cliente Azure OpenAI para funcionalidades avanzadas
        try:
            self.llm_client = iniciar_agente_llm()
        except:
            self.llm_client = None
        
        logger.info(f"🚀 CDG Agent v5.0 inicializado - Modo: {'PRODUCTION' if IMPORTS_SUCCESSFUL else 'FALLBACK'}")

    async def process_request(self, request: CDGRequest) -> CDGResponse:
        """
        Procesa CUALQUIER solicitud del usuario mediante clasificación inteligente
        """
        start_time = datetime.now()
        
        try:
            logger.info(f"🔍 Procesando consulta: '{request.user_message[:100]}...'")
            
            # 1. Clasificar la intención del usuario
            query_type, confidence = await self._classify_user_intent(request.user_message, request.context)
            logger.info(f"📋 Clasificación: {query_type.value} (confianza: {confidence:.2f})")
            
            # 2. Manejo especial para interacciones con gráficos
            if request.chart_interaction_type == 'pivot' and request.current_chart_config:
                return await self._handle_chart_pivot_direct(request, start_time)
            
            # 3. Enriquecer contexto si es necesario
            enriched_context = await self._enrich_context(request, query_type)
            
            # 4. Procesar según el tipo de consulta
            response_content = await self._route_query(request, query_type, enriched_context)
            
            # 5. Generar gráficos si se solicitan
            charts, chart_configs = [], []
            if request.include_charts and query_type != QueryType.CHART_PIVOT:
                charts, chart_configs = await self._generate_charts(response_content, query_type, request)
            
            # 6. Generar recomendaciones si se solicitan
            recommendations = []
            if request.include_recommendations:
                recommendations = await self._generate_recommendations(response_content, query_type)
            
            # 7. Generar sugerencias de pivot para gráficos
            pivot_suggestions = await self._generate_pivot_suggestions(response_content, query_type)
            
            # 8. Construir respuesta final
            execution_time = (datetime.now() - start_time).total_seconds()
            
            response = CDGResponse(
                response_type=query_type,
                content=response_content,
                charts=charts,
                recommendations=recommendations,
                metadata={
                    'user_id': request.user_id,
                    'gestor_id': request.gestor_id,
                    'periodo': request.periodo,
                    'confidence': confidence,
                    'enriched_context': enriched_context,
                    'imports_successful': self.imports_successful,
                    'chart_interaction_type': request.chart_interaction_type
                },
                execution_time=execution_time,
                confidence_score=confidence,
                created_at=datetime.now(),
                chart_configs=chart_configs,
                pivot_suggestions=pivot_suggestions,
                basic_queries_used=request.use_basic_queries,
                data_sources=self._get_data_sources(query_type)
            )
            
            # 9. Actualizar historial
            self._update_conversation_history(request, response)
            
            logger.info(f"✅ Consulta procesada en {execution_time:.2f}s")
            return response
            
        except Exception as e:
            logger.error(f"❌ Error procesando consulta: {e}")
            return await self._create_error_response(request, str(e), start_time)

    async def _classify_user_intent(self, user_message: str, context: Dict[str, Any] = None) -> Tuple[QueryType, float]:
        """
        Clasifica la intención del usuario usando múltiples enfoques:
        1. Patrones de palabras clave
        2. Azure OpenAI con prompts específicos de CDG
        3. Contexto de consultas anteriores
        """
        try:
            # Normalizar mensaje
            message_lower = user_message.lower().strip()
            
            # Clasificación por palabras clave (fallback rápido)
            keyword_classification = self._classify_by_keywords(message_lower)
            
            # Si Azure OpenAI está disponible, usar clasificación avanzada
            if IMPORTS_SUCCESSFUL and self.llm_client:
                try:
                    ai_classification = await self._classify_with_ai(user_message, context or {})
                    # Combinar ambas clasificaciones
                    return self._combine_classifications(keyword_classification, ai_classification)
                except Exception as e:
                    logger.warning(f"Clasificación AI falló, usando keywords: {e}")
            
            return keyword_classification
            
        except Exception as e:
            logger.error(f"Error en clasificación: {e}")
            return QueryType.GENERAL_CHAT, 0.3

    def _classify_by_keywords(self, message_lower: str) -> Tuple[QueryType, float]:
        """Clasificación basada en palabras clave"""
        
        # Patrones específicos con alta confianza
        high_confidence_patterns = [
            # Análisis de gestores
            (r'gestor\s+\d+|gestor\s+[a-zA-Z]+|performance.*gestor|rendimiento.*gestor', 
             QueryType.GESTOR_ANALYSIS, 0.9),
            
            # Comparativas y rankings  
            (r'ranking|comparar|versus|vs|mejor|peor|top\s*\d*|líder|lider', 
             QueryType.COMPARATIVE_ANALYSIS, 0.85),
             
            # Múltiples gestores
            (r'gestores?\s+\d+.*\d+|varios gestores|todos los gestores', 
             QueryType.MULTIPLE_GESTOR_ANALYSIS, 0.9),
             
            # Análisis de centros
            (r'centro|madrid|barcelona|palma|málaga|malaga|bilbao|oficina', 
             QueryType.CENTRO_ANALYSIS, 0.8),
             
            # Precios
            (r'precio.*real|precio.*estándar|precio.*estandar|precio.*producto|pricing', 
             QueryType.PRICING_ANALYSIS, 0.9),
             
            # Desviaciones
            (r'desviación|desviacion|anomalía|anomalia|outlier|alerta', 
             QueryType.DEVIATION_ANALYSIS, 0.85),
             
            # Incentivos
            (r'incentivo|bonus|comisión|comision|variable|salario', 
             QueryType.INCENTIVE_ANALYSIS, 0.85),
             
            # Productos específicos
            (r'banca privada|fondos|hipoteca|depósito|deposito|plazo fijo', 
             QueryType.PRODUCTO_ANALYSIS, 0.8),
             
            # Segmentos
            (r'segmento|banca personal|banca empresas|minorista', 
             QueryType.SEGMENTO_ANALYSIS, 0.8),
             
            # Reportes
            (r'business review|informe|reporte|presentación|presentacion', 
             QueryType.BUSINESS_REVIEW, 0.8),
             
            # KPIs
            (r'kpi|indicador|roe|margen|rentabilidad|eficiencia', 
             QueryType.KPI_SUMMARY, 0.75),
             
            # Consultas rápidas
            (r'resumen|cuántos|cuantos|total|cantidad|lista|listar', 
             QueryType.QUICK_SUMMARY, 0.7),
             
            # Gráficos - detección mejorada
            (r'gráfico|grafico|chart|cambiar.*gráfico|cambiar.*grafico|circular|barras|línea', 
             QueryType.CHART_GENERATION, 0.8),
             
            # Ayuda
            (r'^ayuda|^help|qué puedo|que puedo|cómo funciona|como funciona', 
             QueryType.HELP_REQUEST, 0.9)
        ]
        
        # Buscar patrones de alta confianza
        for pattern, query_type, confidence in high_confidence_patterns:
            if re.search(pattern, message_lower):
                return query_type, confidence
        
        # Patrones de confianza media
        medium_confidence_patterns = [
            (['performance', 'rendimiento', 'resultado'], QueryType.GESTOR_ANALYSIS, 0.6),
            (['datos', 'información', 'info', 'mostrar'], QueryType.DATA_EXPLORATION, 0.5),
            (['periodo', 'mes', 'año', 'trimestre'], QueryType.QUICK_SUMMARY, 0.5),
        ]
        
        for keywords, query_type, confidence in medium_confidence_patterns:
            if any(keyword in message_lower for keyword in keywords):
                return query_type, confidence
        
        # Por defecto: conversación general
        return QueryType.GENERAL_CHAT, 0.4

    async def _classify_with_ai(self, user_message: str, context: Dict[str, Any]) -> Tuple[QueryType, float]:
        """Clasificación avanzada usando Azure OpenAI con prompts específicos de CDG"""
        
        try:
            # Usar el prompt específico para clasificación de intenciones
            classification_prompt = build_intent_classification_prompt(user_message, context)
            
            response = self.llm_client.chat.completions.create(
                model=settings.AZURE_OPENAI_DEPLOYMENT_ID,
                messages=[
                    {"role": "system", "content": "Eres un clasificador experto de intenciones para control de gestión bancario de Banca March."},
                    {"role": "user", "content": classification_prompt}
                ],
                temperature=0.1,
                max_tokens=150
            )
            
            result_text = response.choices[0].message.content.strip()
            
            # Limpiar respuesta JSON
            json_match = re.search(r'\{.*\}', result_text, re.DOTALL)
            if json_match:
                result_text = json_match.group(0)
            
            result = json.loads(result_text)
            query_type = QueryType(result['intent'])
            confidence = float(result['confidence'])
            
            return query_type, confidence
            
        except Exception as e:
            logger.warning(f"Error en clasificación AI: {e}")
            raise

    def _combine_classifications(self, keyword_result: Tuple[QueryType, float], 
                               ai_result: Tuple[QueryType, float]) -> Tuple[QueryType, float]:
        """Combina clasificaciones de keywords y AI"""
        keyword_type, keyword_conf = keyword_result
        ai_type, ai_conf = ai_result
        
        # Si ambas coinciden, usar la confianza más alta
        if keyword_type == ai_type:
            return keyword_type, max(keyword_conf, ai_conf)
        
        # Si AI tiene mayor confianza, usar AI
        if ai_conf > keyword_conf + 0.2:
            return ai_type, ai_conf
        
        # Caso contrario, usar keywords (más estable)
        return keyword_type, keyword_conf

    async def _handle_chart_pivot_direct(self, request: CDGRequest, start_time: datetime) -> CDGResponse:
        """Manejo directo de pivoteo de gráficos usando el chart_generator mejorado"""
        try:
            logger.info("🔄 Procesando pivoteo de gráfico")
            
            # Usar la nueva función integrada del chart_generator
            pivot_result = handle_chart_change_request(
                request.user_message,
                request.current_chart_config or {},
                {
                    'agente': 'CDG',
                    'user_id': request.user_id,
                    'gestor_id': request.gestor_id,
                    'periodo': request.periodo
                }
            )
            
            execution_time = (datetime.now() - start_time).total_seconds()
            
            # Si el pivoteo fue exitoso, regenerar el gráfico con la nueva configuración
            charts = []
            if pivot_result.get('status') == 'success' and pivot_result.get('new_config'):
                try:
                    # Regenerar gráfico con nueva configuración si tenemos datos
                    if 'data' in pivot_result:
                        new_chart = create_chart_from_query_data(
                            pivot_result['data'],
                            pivot_result['new_config']
                        )
                        if new_chart.get('id'):
                            charts.append(new_chart)
                except Exception as chart_error:
                    logger.warning(f"Error regenerando gráfico: {chart_error}")
            
            return CDGResponse(
                response_type=QueryType.CHART_PIVOT,
                content={
                    'pivot_result': pivot_result,
                    'message': pivot_result.get('message', 'Gráfico modificado'),
                    'changes_made': pivot_result.get('changes_made', []),
                    'new_config': pivot_result.get('new_config', {}),
                    'method': pivot_result.get('interpretation_method', 'pattern_matching')
                },
                charts=charts,
                execution_time=execution_time,
                confidence_score=0.9,  # Alto para operaciones de pivoteo exitosas
                created_at=datetime.now(),
                data_sources=['chart_generator', 'Azure_OpenAI'] if IMPORTS_SUCCESSFUL else ['chart_generator']
            )
            
        except Exception as e:
            logger.error(f"Error en pivoteo directo: {e}")
            return await self._create_error_response(request, f"Error modificando gráfico: {str(e)}", start_time)

    async def _generate_charts(self, content: Dict[str, Any], query_type: QueryType, request: CDGRequest) -> Tuple[List[Dict], List[Dict]]:
        """Genera gráficos inteligentes basados en el contenido y tipo de consulta"""
        charts = []
        configs = []
        
        try:
            if query_type == QueryType.GESTOR_ANALYSIS and 'gestor_data' in content:
                # Usar el nuevo generador integrado para gestores
                chart = self.chart_generator.generate_gestor_dashboard(
                    content['gestor_data'], 
                    content.get('kpis', {}),
                    content.get('periodo')
                )
                if chart and chart.get('charts'):
                    charts.extend(chart['charts'])
                    
            elif query_type in [QueryType.COMPARATIVE_ANALYSIS, QueryType.RANKING_QUERY]:
                # Usar QueryIntegratedChartGenerator para rankings
                if 'ranking_gestores' in content and content['ranking_gestores']:
                    ranking_chart = self.query_chart_generator.generate_gestores_ranking_chart(
                        metric='CONTRATOS',
                        chart_type='horizontal_bar'
                    )
                    if ranking_chart.get('id'):
                        charts.append(ranking_chart)
                        configs.append(ranking_chart.get('config', {}))
                        
            elif query_type == QueryType.PRICING_ANALYSIS:
                # Gráfico específico para análisis de precios
                pricing_chart = self.query_chart_generator.generate_precios_comparison_chart(
                    fecha_calculo=request.periodo,
                    chart_type='bar'
                )
                if pricing_chart.get('id'):
                    charts.append(pricing_chart)
                    configs.append(pricing_chart.get('config', {}))
                    
            elif query_type == QueryType.QUICK_SUMMARY:
                # Dashboard resumen usando el generador integrado
                summary_dashboard = self.query_chart_generator.generate_summary_dashboard()
                if summary_dashboard.get('charts'):
                    charts.extend(summary_dashboard['charts'])
                    
            elif query_type == QueryType.CHART_GENERATION:
                # Generar gráfico basado en datos disponibles en content
                chart_data = self._extract_chart_data_from_content(content)
                if chart_data:
                    chart_config = {
                        'chart_type': 'bar',
                        'dimension': 'gestor',
                        'metric': 'PERFORMANCE'
                    }
                    generated_chart = create_chart_from_query_data(chart_data, chart_config)
                    if generated_chart.get('id'):
                        charts.append(generated_chart)
                        configs.append(chart_config)
                        
        except Exception as e:
            logger.warning(f"Error generando gráficos: {e}")
        
        return charts, configs

    def _extract_chart_data_from_content(self, content: Dict[str, Any]) -> List[Dict[str, Any]]:
        """Extrae datos apropiados para gráficos del contenido de respuesta"""
        chart_data = []
        
        try:
            # Buscar diferentes tipos de datos estructurados
            if 'gestor_data' in content:
                gestor_data = content['gestor_data']
                if isinstance(gestor_data, dict):
                    chart_data.append({
                        'label': gestor_data.get('DESC_GESTOR', 'Gestor'),
                        'value': gestor_data.get('margen_neto_pct', gestor_data.get('performance', 0)),
                        'original_data': gestor_data
                    })
                    
            elif 'ranking_gestores' in content:
                ranking = content['ranking_gestores']
                if isinstance(ranking, list):
                    for item in ranking[:10]:  # Top 10
                        chart_data.append({
                            'label': item.get('DESC_GESTOR', f"Gestor {item.get('gestor_id', '')}"),
                            'value': item.get('margen_neto_pct', item.get('performance', 0)),
                            'original_data': item
                        })
                        
            elif 'comparisons_by_segment' in content:
                segments = content['comparisons_by_segment']
                if isinstance(segments, dict):
                    for segment_id, comparisons in segments.items():
                        if comparisons:
                            chart_data.append({
                                'label': segment_id,
                                'value': len(comparisons),
                                'original_data': {'segment': segment_id, 'count': len(comparisons)}
                            })
                            
        except Exception as e:
            logger.warning(f"Error extrayendo datos para gráfico: {e}")
        
        return chart_data

    async def _generate_natural_response(self, content: Dict[str, Any], query_type: QueryType, user_context: Dict[str, Any]) -> str:
        """Genera respuesta conversacional natural usando los prompts específicos"""
        try:
            if not IMPORTS_SUCCESSFUL or not self.llm_client:
                return self._generate_fallback_response(content, query_type)
            
            # Usar el prompt específico para respuesta natural
            response_prompt = build_natural_response_prompt(content, user_context)
            
            response = self.llm_client.chat.completions.create(
                model=settings.AZURE_OPENAI_DEPLOYMENT_ID,
                messages=[
                    {"role": "system", "content": CHAT_NATURAL_RESPONSE_SYSTEM_PROMPT},
                    {"role": "user", "content": response_prompt}
                ],
                temperature=0.7,
                max_tokens=1200
            )
            
            return response.choices[0].message.content.strip()
            
        except Exception as e:
            logger.warning(f"Error generando respuesta natural: {e}")
            return self._generate_fallback_response(content, query_type)

    def _generate_fallback_response(self, content: Dict[str, Any], query_type: QueryType) -> str:
        """Genera respuesta fallback cuando no está disponible Azure OpenAI"""
        if 'error' in content:
            return f"Se produjo un error: {content['error']}"
        
        response_templates = {
            QueryType.GESTOR_ANALYSIS: "Análisis del gestor {gestor_id} completado.",
            QueryType.COMPARATIVE_ANALYSIS: "Análisis comparativo realizado.",
            QueryType.PRICING_ANALYSIS: "Análisis de precios completado.",
            QueryType.QUICK_SUMMARY: "Resumen del sistema generado.",
            QueryType.HELP_REQUEST: "Sistema CDG disponible para consultas de control de gestión."
        }
        
        template = response_templates.get(query_type, "Consulta procesada correctamente.")
        
        # Sustituir variables si están disponibles
        if '{gestor_id}' in template and 'gestor_id' in content:
            template = template.replace('{gestor_id}', str(content['gestor_id']))
        
        return template

    async def _handle_gestor_analysis(self, request: CDGRequest, context: Dict[str, Any]) -> Dict[str, Any]:
        """Análisis específico de gestor con respuesta natural integrada"""
        try:
            gestor_id = request.gestor_id or self._extract_gestor_id_from_message(request.user_message)
            periodo = request.periodo or context.get('periodo_actual', '2025-10')
            
            if not gestor_id:
                return {
                    'error': 'No se pudo identificar el gestor. Especifica un ID de gestor.',
                    'gestores_disponibles': context.get('gestores_disponibles', [])
                }
            
            # Obtener datos del gestor
            gestor_data = self.gestor_queries.get_gestor_performance_enhanced(gestor_id, periodo)
            
            if not gestor_data or not hasattr(gestor_data, 'data') or not gestor_data.data:
                return {'error': f'No se encontraron datos para el gestor {gestor_id}'}
            
            gestor_info = gestor_data.data[0]
            
            # Calcular KPIs
            kpis = self.kpi_calculator.calculate_kpis_from_data(gestor_info)
            
            # Obtener comparativas
            ranking = self.comparative_queries.ranking_gestores_por_margen_enhanced(periodo)
            
            result = {
                'gestor_id': gestor_id,
                'periodo': periodo,
                'gestor_data': gestor_info,
                'kpis': kpis,
                'ranking_position': self._find_gestor_position(gestor_id, ranking),
                'analysis_type': 'gestor_individual'
            }
            
            # Generar respuesta natural si no es modo rápido
            if not request.quick_mode:
                user_context = {
                    'gestor_id': gestor_id,
                    'periodo': periodo,
                    'user_level': 'intermedio'  # Por defecto
                }
                natural_response = await self._generate_natural_response(result, QueryType.GESTOR_ANALYSIS, user_context)
                result['natural_response'] = natural_response
            
            return result
            
        except Exception as e:
            logger.error(f"Error en análisis de gestor: {e}")
            return {'error': str(e)}

    async def _handle_pricing_analysis(self, request: CDGRequest, context: Dict[str, Any]) -> Dict[str, Any]:
        """Análisis específico de precios"""
        try:
            # Detectar si la consulta es sobre banca privada específicamente
            message_lower = request.user_message.lower()
            
            if 'banca privada' in message_lower or 'privada' in message_lower:
                # Consulta específica de banca privada
                if request.use_basic_queries:
                    # Obtener precios de banca privada usando basic_queries
                    std_prices = self.basic_queries.get_precios_std_by_segmento('N10102')  # N10102 = Banca Privada
                    real_prices = self.basic_queries.get_precios_real_by_segmento_periodo('N10102', '2025-10')
                    
                    # Comparación
                    comparison = self.basic_queries.compare_precios_std_vs_real('2025-10')
                    banca_privada_comparison = [item for item in comparison if item.get('SEGMENTOID') == 'N10102']
                    
                    return {
                        'segmento': 'Banca Privada',
                        'segmento_id': 'N10102',
                        'precios_std': std_prices,
                        'precios_real': real_prices,
                        'comparison': banca_privada_comparison,
                        'analysis_type': 'pricing_banca_privada'
                    }
            
            # Análisis general de precios
            periodo = request.periodo or context.get('periodo_actual', '2025-10')
            
            # Obtener comparaciones de precios por todos los segmentos
            all_comparisons = self.basic_queries.compare_precios_std_vs_real(periodo)
            
            # Agrupar por segmento
            by_segment = {}
            for item in all_comparisons:
                segment_id = item.get('SEGMENTOID', 'Unknown')
                if segment_id not in by_segment:
                    by_segment[segment_id] = []
                by_segment[segment_id].append(item)
            
            return {
                'periodo': periodo,
                'comparisons_by_segment': by_segment,
                'all_comparisons': all_comparisons,
                'analysis_type': 'pricing_general'
            }
            
        except Exception as e:
            logger.error(f"Error en análisis de precios: {e}")
            return {'error': str(e)}

    def _extract_gestor_id_from_message(self, message: str) -> Optional[str]:
        """Extrae ID de gestor del mensaje"""
        # Buscar patrones como "gestor 15", "gestor 3", etc.
        match = re.search(r'gestor\s+(\d+)', message.lower())
        if match:
            return match.group(1)
        
        # Buscar números sueltos que puedan ser IDs
        numbers = re.findall(r'\b(\d+)\b', message)
        if numbers:
            # Asumir que el primer número válido (1-30) es el gestor_id
            for num in numbers:
                if 1 <= int(num) <= 30:
                    return num
        
        return None

    def _find_gestor_position(self, gestor_id: str, ranking_data) -> Optional[int]:
        """Encuentra la posición del gestor en el ranking"""
        try:
            if ranking_data and hasattr(ranking_data, 'data'):
                for i, gestor in enumerate(ranking_data.data, 1):
                    if str(gestor.get('gestor_id', '')) == str(gestor_id):
                        return i
        except:
            pass
        return None

# Funciones de conveniencia para integración

def create_cdg_agent() -> CDGAgent:
    """Crea una instancia del agente CDG"""
    return CDGAgent()

async def process_cdg_request(user_message: str, **kwargs) -> Dict[str, Any]:
    """
    Función de conveniencia para procesar una consulta
    
    Args:
        user_message: Mensaje del usuario
        **kwargs: Parámetros adicionales (gestor_id, periodo, etc.)
    """
    agent = create_cdg_agent()
    
    request = CDGRequest(
        user_message=user_message,
        **kwargs
    )
    
    response = await agent.process_request(request)
    return response.to_dict()

async def process_quick_query(user_message: str, **kwargs) -> Dict[str, Any]:
    """Procesamiento rápido con basic_queries"""
    return await process_cdg_request(
        user_message,
        quick_mode=True,
        use_basic_queries=True,
        include_charts=False,
        **kwargs
    )
