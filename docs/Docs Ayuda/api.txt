/**
 * CDG Frontend API Client (para FastAPI v6.2)
 * --------------------------------------------
 * - Desenvelope automático: devuelve { data, meta, ts } por defecto.
 * - Retries con backoff para 429/502/503/504.
 * - Cancelación de requests (AbortController).
 * - WS helper para chat: /chat/ws/{user_id}.
 * - SDK por módulos alineado con main.py (v6.2).
 */

import axios from "axios";

/* ============================
 * Configuración base
 * ============================ */

const BASE_URL =
  process.env.REACT_APP_API_BASE_URL?.replace(/\/+$/, "") ||
  // fallback razonable en local
  `${window.location.protocol}//${window.location.hostname}:8000`;

const DEFAULT_TIMEOUT =
  Number(process.env.REACT_APP_API_TIMEOUT) > 0
    ? Number(process.env.REACT_APP_API_TIMEOUT)
    : 20000; // 20s

// Rutas conocidas que aceptan GET con querystring y POST con JSON
const JSON_HEADERS = {
  "Content-Type": "application/json",
  Accept: "application/json",
};

// Helper: construir qs limpio
const buildQuery = (params = {}) => {
  const esc = encodeURIComponent;
  const qs = Object.entries(params)
    .filter(([, v]) => v !== undefined && v !== null && v !== "")
    .map(([k, v]) => `${esc(k)}=${esc(v)}`)
    .join("&");
  return qs ? `?${qs}` : "";
};

// Error normalizado para el front
class ApiClientError extends Error {
  constructor(message, { status = 0, code = 0, detail = null } = {}) {
    super(message);
    this.name = "ApiClientError";
    this.status = status;
    this.code = code || status;
    this.detail = detail;
  }
}

/* ============================
 * Axios instance + interceptors
 * ============================ */

const instance = axios.create({
  baseURL: BASE_URL,
  timeout: DEFAULT_TIMEOUT,
  headers: JSON_HEADERS,
  // Envío de cookies si en el futuro servimos desde el mismo dominio con auth
  withCredentials: false,
});

// Exponential backoff simple
const sleep = (ms) => new Promise((r) => setTimeout(r, ms));
const shouldRetry = (error) => {
  if (!error || !error.response) return false;
  const { status } = error.response;
  return [429, 502, 503, 504].includes(status);
};

// Interceptor de respuesta: desenvelope + retry
instance.interceptors.response.use(
  (res) => {
    // Esperamos {status, data, meta, timestamp}
    const payload = res.data;
    if (payload && typeof payload === "object" && "status" in payload) {
      if (payload.status !== "success") {
        // Back se encarga de uniformar errores graves vía handlers, por si acaso:
        throw new ApiClientError(payload.message || "Error API", {
          status: res.status,
          code: payload.code || res.status,
          detail: payload,
        });
      }
      return {
        data: payload.data,
        meta: payload.meta || null,
        ts: payload.timestamp || null,
        raw: payload,
      };
    }
    // Si algún endpoint devolviera algo crudo (no debería), entregamos tal cual
    return { data: payload, meta: null, ts: null, raw: payload };
  },
  async (error) => {
    // Reintentos automáticos
    const config = error.config || {};
    config.__retryCount = config.__retryCount || 0;

    if (shouldRetry(error) && config.__retryCount < 3) {
      config.__retryCount += 1;
      const backoff = 400 * 2 ** (config.__retryCount - 1); // 400ms, 800ms, 1600ms
      await sleep(backoff);
      return instance(config);
    }

    // Normalización de error para el front
    const status = error.response?.status || 0;
    const backend = error.response?.data;
    const messageFromBackend =
      backend?.message ||
      backend?.detail ||
      backend?.error ||
      error.message ||
      "Error de red";

    throw new ApiClientError(messageFromBackend, {
      status,
      code: backend?.code || status,
      detail: backend || null,
    });
  }
);

/* ============================
 * Core HTTP helpers
 * ============================ */

const withAbort = (config = {}) => {
  // Permite pasar signal desde componentes
  if (config.signal) return config;
  const controller = new AbortController();
  return { ...config, signal: controller.signal, __controller: controller };
};

const http = {
  get: (url, { params, ...cfg } = {}) =>
    instance.get(`${url}${buildQuery(params)}`, withAbort(cfg)),
  post: (url, body = {}, cfg = {}) =>
    instance.post(url, body, withAbort(cfg)),
  put: (url, body = {}, cfg = {}) => instance.put(url, body, withAbort(cfg)),
  del: (url, cfg = {}) => instance.delete(url, withAbort(cfg)),
  // acceso al axios por si hace falta
  raw: instance,
  baseURL: BASE_URL,
};

/* ============================
 * Helpers utilitarios
 * ============================ */

const unwrap = async (promise, { returnMeta = false } = {}) => {
  const { data, meta } = await promise;
  return returnMeta ? { data, meta } : data;
};

const toQueryBody = (obj) =>
  Object.fromEntries(
    Object.entries(obj || {}).filter(
      ([, v]) => v !== undefined && v !== null && v !== ""
    )
  );

/* ============================
 * WS Chat helper

const makeWsUrl = (path) => {
  const base = BASE_URL.replace(/^http/, 'ws').replace(/\/$/, '');
  return `${base}${path}`;
};

/**
 * Abre un WebSocket con el chat.
 * Devuelve { socket, sendJson, close }.
 */
// En api.js, línea ~190
// ✅ CORRECCIÓN: Heartbeat menos agresivo
const openChatSocket = (userId, { onMessage, onOpen, onClose, onError } = {}) => {
  if (!userId) throw new Error("userId requerido para WS");
  const url = makeWsUrl(`/chat/ws/${encodeURIComponent(userId)}`);
  
  console.log(`[WS] Conectando a: ${url}`);
  const socket = new WebSocket(url);
  
  // ✅ CORRECCIÓN: Heartbeat cada 45s (más conservador)
  let heartbeatInterval;
  let isAlive = false;

  // ✅ CONFIGURACIÓN ESPECÍFICA PARA UVICORN WINDOWS
  socket.onopen = (evt) => {
    console.log('[WS] ✅ Conectado exitosamente');
    isAlive = true;

    // ✅ Heartbeat más conservador para Uvicorn Windows
    heartbeatInterval = setInterval(() => {
      if (socket.readyState === WebSocket.OPEN) {
        console.log('[WS] 💓 Enviando heartbeat');
        socket.send(JSON.stringify({ type: 'ping', ts: Date.now() }));

        // ✅ Timeout de 20s (compatible con configuración del servidor)
        isAlive = false;
        setTimeout(() => {
          if (!isAlive && socket.readyState === WebSocket.OPEN) {
            console.log('[WS] ❌ No recibió pong, cerrando conexión');
            socket.close(1000, 'Ping timeout'); 
          }
        }, 20000); // 20 segundos
      }
    }, 35000); // ✅ Cada 35 segundos (menos frecuente que el servidor)

    onOpen && onOpen(evt);
  };


  
  socket.onmessage = (evt) => {
    try {
      const msg = JSON.parse(evt.data);
      
      // Manejar pong del servidor
      if (msg.type === 'pong') {
        console.log('[WS] 💚 Recibido pong del servidor');
        isAlive = true;
        return;
      }
      
      onMessage && onMessage(msg);
    } catch (e) {
      console.warn('[WS] Parse error:', e);
    }
  };
  
  socket.onerror = (evt) => {
    console.error('[WS] ❌ Error:', evt);
    onError && onError(evt);
  };
  
  socket.onclose = (evt) => {
    console.log(`[WS] 🔌 Cerrado: ${evt.code} - ${evt.reason || 'Sin razón'}`);
    
    // Limpiar heartbeat
    if (heartbeatInterval) {
      clearInterval(heartbeatInterval);
      heartbeatInterval = null;
    }
    
    onClose && onClose(evt);
  };

  const sendJson = (obj) => {
    if (socket.readyState === WebSocket.OPEN) {
      socket.send(JSON.stringify(obj));
    } else {
      console.warn('[WS] ⚠️ Cannot send, socket not open:', socket.readyState);
    }
  };

  const close = () => {
    if (heartbeatInterval) {
      clearInterval(heartbeatInterval);
      heartbeatInterval = null;
    }
    socket.close();
  };

  return { socket, sendJson, close };
};
