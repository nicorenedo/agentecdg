/**
 * Helper para formatear valores bancarios
 */
const formatBankingValue = (value, options = {}) => {
  const { showCurrency = true, decimals = 0 } = options;
  if (typeof value !== 'number') return { displayValue: '0', isNegative: false };
  const abs = Math.abs(value);
  let display = '';
  if (abs >= 1000000) display = (abs / 1000000).toFixed(decimals) + 'M';
  else if (abs >= 1000) display = (abs / 1000).toFixed(decimals) + 'K';
  else display = abs.toFixed(decimals);
  return { 
    displayValue: showCurrency ? display + '€' : display, 
    isNegative: value < 0, 
    originalValue: value, 
    numericValue: abs 
  };
};

/**
 * Helper para obtener labels de dataset según la métrica
 */
const getDatasetLabelFromMetric = (metric) => {
  const labels = {
    'CONTRATOS': 'Contratos',
    'CLIENTES': 'Clientes', 
    'INGRESOS': 'Ingresos (€)',
    'MARGEN': 'Margen (%)',
    'VOLUMEN': 'Volumen'
  };
  return labels[metric] || metric;
};

/**
 * Validador mejorado que no confunde "cargando" con "vacío"
 */
const validateDataset = (dataset, context = '') => {
  if (!dataset) {
    console.log(`[${context}] Waiting for dataset...`);
    return { valid: false, isEmpty: false, isLoading: true };
  }
  
  if (dataset.labels && Array.isArray(dataset.labels) && dataset.datasets && Array.isArray(dataset.datasets)) {
    const hasData = dataset.labels.length > 0 && 
                   dataset.datasets.some(ds => ds.data && ds.data.some(val => val !== null && val !== undefined && val > 0));
    
    if (!hasData) {
      console.log(`[${context}] Dataset structure valid but no meaningful data`);
      return { valid: true, isEmpty: true, isLoading: false };
    }
    console.log(`[${context}] ✅ Valid dataset with data`);
    return { valid: true, isEmpty: false, isLoading: false };
  }
  
  if (Array.isArray(dataset)) {
    if (dataset.length === 0) {
      console.log(`[${context}] Empty array dataset`);
      return { valid: true, isEmpty: true, isLoading: false };
    }
    return { valid: true, isEmpty: false, isLoading: false };
  }
  
  return { valid: false, isEmpty: true, isLoading: false };
};

/**
 * SimpleChart con gráficos reales y tooltips avanzados
 */
const SimpleChart = ({ data, config = {}, chartKey, height = 260, onDataClick }) => {
  const { chartType = 'horizontal_bar' } = config;
  
  const validation = validateDataset(data, chartKey);
  
  if (validation.isLoading) {
    return (
      <div style={{ 
        height, 
        display: 'flex', 
        flexDirection: 'column', 
        justifyContent: 'center', 
        alignItems: 'center' 
      }}>
        <Loader size="large" />
        <Text type="secondary" style={{ marginTop: 16 }}>
          Cargando datos del gráfico...
        </Text>
      </div>
    );
  }
  
  if (!validation.valid || validation.isEmpty) {
    return (
      <div style={{ 
        height, 
        display: 'flex', 
        flexDirection: 'column', 
        justifyContent: 'center', 
        alignItems: 'center' 
      }}>
        <Empty 
          image={<BarChartOutlined style={{ fontSize: 48, color: theme.colors?.textSecondary || '#999', opacity: 0.3 }} />}
          description={
            <div style={{ textAlign: 'center' }}>
              <Text type="secondary">No hay datos disponibles</Text>
              <br />
              <Text type="secondary" style={{ fontSize: 11 }}>
                Intenta cambiar el período o gestor
              </Text>
            </div>
          }
        />
      </div>
    );
  }

  const commonOptions = {
    responsive: true,
    maintainAspectRatio: false,
    plugins: {
      legend: {
        position: 'top',
        display: true,
        labels: {
          usePointStyle: true,
          padding: 20,
          color: theme.colors?.textPrimary || '#333',
          font: { size: 12 }
        }
      },
      tooltip: {
        enabled: true,
        mode: 'nearest',
        intersect: false,
        backgroundColor: 'rgba(0,0,0,0.8)',
        titleColor: '#fff',
        bodyColor: '#fff',
        borderColor: theme.colors?.bmGreenPrimary || '#1890ff',
        borderWidth: 1,
        cornerRadius: 6,
        displayColors: true,
        callbacks: {
          title: (context) => context[0].label || '',
          label: (context) => {
            const value = context.parsed.y || context.parsed;
            const formatted = formatBankingValue(Math.abs(value), { decimals: 2 });
            return ` ${context.dataset.label || ''}: ${formatted.displayValue}`;
          }
        }
      }
    },
    onClick: (event, elements) => {
      if (elements.length > 0 && onDataClick) {
        const clickedElement = elements[0];
        const dataIndex = clickedElement.index;
        onDataClick({
          index: dataIndex,
          value: data.datasets[0].data[dataIndex],
          label: data.labels[dataIndex],
          chartKey
        });
      }
    },
    layout: {
      padding: { top: 10, bottom: 10, left: 10, right: 10 }
    }
  };

  const renderChart = () => {
    console.log(`[${chartKey}] Rendering ${chartType} with data:`, data);
    
    switch (chartType) {
      case 'horizontal_bar':
        return (
          <Bar 
            data={data} 
            options={{
              ...commonOptions,
              indexAxis: 'y',
              scales: {
                x: {
                  beginAtZero: true,
                  grid: { color: 'rgba(0,0,0,0.1)', drawBorder: false },
                  ticks: {
                    color: theme.colors?.textSecondary || '#666',
                    font: { size: 11 },
                    callback: function(value) {
                      return formatBankingValue(value, { decimals: 0 }).displayValue;
                    }
                  }
                },
                y: {
                  grid: { display: false },
                  ticks: {
                    color: theme.colors?.textPrimary || '#333',
                    font: { size: 12, weight: '500' },
                    maxRotation: 0,
                    callback: function(value, index) {
                      const label = this.getLabelForValue(value);
                      return label.length > 20 ? label.substring(0, 17) + '...' : label;
                    }
                  }
                }
              }
            }}
            height={height}
          />
        );
        
      case 'bar':
      case 'comparison':
        return (
          <Bar 
            data={data} 
            options={{
              ...commonOptions,
              plugins: {
                ...commonOptions.plugins,
                title: {
                  display: !!(data.meta?.segmentoNombre),
                  text: data.meta?.segmentoNombre ? `Segmento: ${data.meta.segmentoNombre}` : '',
                  font: { size: 14, weight: 'bold' },
                  color: theme.colors?.textPrimary || '#333',
                  padding: { bottom: 20 }
                },
                legend: {
                  ...commonOptions.plugins.legend,
                  position: 'bottom',
                  labels: {
                    usePointStyle: true,
                    padding: 15,
                    font: { size: 13, weight: '500' }
                  }
                },
                tooltip: {
                  ...commonOptions.plugins.tooltip,
                  callbacks: {
                    title: (context) => `${context[0].label}`,
                    label: (context) => {
                      const value = context.parsed.y;
                      const formatted = formatBankingValue(value, { decimals: 2 });
                      return ` ${context.dataset.label}: ${formatted.displayValue}`;
                    },
                    afterBody: (context) => {
                      if (context.length < 2) return [];
                      
                      const chart = context[0].chart;
                      const dataIndex = context[0].dataIndex;
                      
                      const standardData = chart.data.datasets.find(ds => ds.label.includes('Estándar'));
                      const realData = chart.data.datasets.find(ds => ds.label.includes('Real'));
                      
                      if (standardData && realData) {
                        const standardValue = standardData.data[dataIndex];
                        const realValue = realData.data[dataIndex];
                        const difference = realValue - standardValue;
                        const percentage = standardValue > 0 ? ((difference / standardValue) * 100) : 0;
                        
                        return [
                          '',
                          `Diferencia: ${formatBankingValue(Math.abs(difference), { decimals: 2 }).displayValue}`,
                          `Variación: ${percentage.toFixed(1)}%`,
                          `Estado: ${Math.abs(percentage) <= 2 ? '✅ Dentro del rango' : Math.abs(percentage) <= 15 ? '⚠️ Atención' : '🚨 Fuera de rango'}`
                        ];
                      }
                      return [];
                    }
                  }
                }
              },
              scales: {
                y: {
                  beginAtZero: true,
                  grid: { color: 'rgba(0,0,0,0.1)', drawBorder: false },
                  ticks: {
                    color: theme.colors?.textSecondary || '#666',
                    font: { size: 11 },
                    callback: function(value) {
                      return formatBankingValue(Math.abs(value), { decimals: 0 }).displayValue;
                    }
                  },
                  title: {
                    display: true,
                    text: 'Precio por Contrato (€)',
                    color: theme.colors?.textPrimary || '#333',
                    font: { size: 12, weight: '600' }
                  }
                },
                x: {
                  grid: { display: false },
                  ticks: {
                    color: theme.colors?.textPrimary || '#333',
                    font: { size: 12, weight: '500' },
                    maxRotation: 45,
                    callback: function(value, index) {
                      return this.getLabelForValue(value);
                    }
                  }
                }
              }
            }}
            height={height}
          />
        );
        
      case 'donut':
      case 'pie':
        return (
          <Doughnut 
            data={data} 
            options={{
              ...commonOptions,
              cutout: chartType === 'donut' ? '60%' : '0%',
              plugins: {
                ...commonOptions.plugins,
                legend: {
                  ...commonOptions.plugins.legend,
                  position: 'right',
                  align: 'center'
                },
                tooltip: {
                  ...commonOptions.plugins.tooltip,
                  callbacks: {
                    ...commonOptions.plugins.tooltip.callbacks,
                    label: (context) => {
                      const value = context.parsed;
                      const total = context.dataset.data.reduce((a, b) => a + b, 0);
                      const percentage = ((value / total) * 100).toFixed(1);
                      return ` ${context.label}: ${percentage}% (${value})`;
                    }
                  }
                }
              }
            }}
            height={height}
          />
        );
        
      default:
        return (
          <Bar 
            data={data} 
            options={commonOptions}
            height={height}
          />
        );
    }
  };

  return (
    <div style={{ height, width: '100%', position: 'relative' }}>
      {renderChart()}
    </div>
  );
};

SimpleChart.propTypes = {

  config: PropTypes.object,
  chartKey: PropTypes.string.isRequired,
  height: PropTypes.number,
  onDataClick: PropTypes.func

/**
 * ✅ InteractiveCharts - Componente principal COMPLETAMENTE CORREGIDO
 * CORRECCIÓN PRINCIPAL: Eliminar protección de carga global que evitaba cargas individuales
 */
const InteractiveCharts = ({
  mode = 'direccion',
  periodo,
  gestorId = null,
  metric = 'CONTRATOS',
  chartType = null,
  filters = {},
  height = 340,
  onReload = () => {},
  onSelectEntity = () => {},
  externalChartConfig = null,
  onChartConfigChange = null,
  className = '',
  style = {}
}) => {
  
  // ✅ NORMALIZACIÓN ESTABLE DE PROPS CON USEMEMO
  const normalizedPeriodo = useMemo(() => {
    if (!periodo) return '2025-10';
    if (typeof periodo === 'string') return periodo;
    if (typeof periodo === 'object') {
      return periodo.latest || periodo.value || periodo.periodo || '2025-10';
    }
    return String(periodo);
  }, [periodo]);

  const normalizedMetric = useMemo(() => metric ? metric.toUpperCase() : 'CONTRATOS', [metric]);

  // ✅ MEMOIZACIÓN PROFUNDA DE FILTROS PARA EVITAR OBJETOS NUEVOS EN CADA RENDER
  const stableFilters = useMemo(() => {
    const filterKeys = Object.keys(filters).sort();
    const stableObj = {};
    filterKeys.forEach(key => {
      stableObj[key] = filters[key];
    });
    return stableObj;
  }, [filters]);

  // ✅ CLAVE ÚNICA PARA DETECTAR CAMBIOS REALES EN DEPENDENCIAS
  const dependencyKey = useMemo(() => 
    JSON.stringify({
      mode,
      gestorId,
      periodo: normalizedPeriodo,
      metric: normalizedMetric,
      chartType,
      filters: stableFilters
    }), 
    [mode, gestorId, normalizedPeriodo, normalizedMetric, chartType, stableFilters]
  );

  // ✅ ESTADO INICIAL LIMPIO
  const [chartsData, setChartsData] = useState({});
  const [loadingStates, setLoadingStates] = useState({
    ranking: false,
    distribution: false,
    priceCompare: false
  });
  const [errorStates, setErrorStates] = useState({});
  const [selectedEntity, setSelectedEntity] = useState(null);

  // ✅ REFS PARA CONTROL DE BUCLES Y ABORT CONTROLLERS
  const abortControllers = useRef({});
  const lastDependencyKey = useRef(null);

  // ✅ CLEANUP MEJORADO PARA ABORT CONTROLLERS
  const cleanupAbortController = useCallback((key) => {
    if (abortControllers.current[key]) {
      try {
        abortControllers.current[key].abort();
      } catch (e) {
        console.warn(`[${key}] Error aborting controller:`, e);
      }
      delete abortControllers.current[key];
    }
  }, []);

  // ✅ FUNCIÓN GENÉRICA DE CARGA SIN PROTECCIÓN GLOBAL - ESTA ES LA CORRECCIÓN CLAVE
  const loadChartData = useCallback(async (chartKey, loaderFn, shouldLoad = true) => {
    // Protección 1: No cargar si no se debe
    if (!shouldLoad) {
      console.log(`[${chartKey}] ⏭️ Skipping load (shouldLoad=false)`);
      return;
    }

    // ⭐ CAMBIO CLAVE: REMOVER PROTECCIÓN GLOBAL QUE CAUSABA EL PROBLEMA
    // ❌ CÓDIGO REMOVIDO: 
    // if (isLoadingRef.current) {
    //   console.log(`[${chartKey}] ⚠️ Already loading globally, skipping...`);
    //   return;
    // }

    // Cleanup de controller previo
    cleanupAbortController(chartKey);
    abortControllers.current[chartKey] = new AbortController();
    
    console.log(`[${chartKey}] 🚀 Loading data...`);
    setLoadingStates(prev => ({ ...prev, [chartKey]: true }));
    setErrorStates(prev => ({ ...prev, [chartKey]: null }));
    
    try {
      const result = await loaderFn();
      
      // Verificar que no se abortó la request
      if (!abortControllers.current[chartKey]?.signal.aborted) {
        console.log(`[${chartKey}] ✅ Data loaded successfully:`, result);
        setChartsData(prev => ({ ...prev, [chartKey]: result }));
      }
      
    } catch (error) {
      if (error.name !== 'AbortError' && !abortControllers.current[chartKey]?.signal.aborted) {
        console.error(`[${chartKey}] ❌ Error loading data:`, error);
        setErrorStates(prev => ({ ...prev, [chartKey]: error }));
      }
    } finally {
      if (!abortControllers.current[chartKey]?.signal.aborted) {
        setLoadingStates(prev => ({ ...prev, [chartKey]: false }));
      }
    }
  }, [cleanupAbortController]);

  // ✅ FUNCIONES DE CARGA ESPECÍFICAS MEMOIZADAS CON DEPENDENCIAS ESTABLES
  const loadDirectionRanking = useCallback(async () => {
    return await analyticsService.getDirectionChartData('gestores-ranking', {
      metric: normalizedMetric,
      chart_type: chartType || 'horizontal_bar',
      periodo: normalizedPeriodo,
      ...stableFilters
    });
  }, [normalizedMetric, normalizedPeriodo, stableFilters, chartType]);

  const loadDirectionDistribution = useCallback(async () => {
    return await analyticsService.getDirectionChartData('centros-distribution', {
      chart_type: chartType || 'donut',
      periodo: normalizedPeriodo,
      ...stableFilters
    });
  }, [normalizedPeriodo, stableFilters, chartType]);

  const loadGestorRanking = useCallback(async () => {
    if (!gestorId) return null;
    return await analyticsService.getTopClientsChartData(gestorId, { 
      limit: 10,
      ...stableFilters
    });
  }, [gestorId, stableFilters]);

  const loadGestorDistribution = useCallback(async () => {
    if (!gestorId) return null;
    return await analyticsService.getProductMixChartData(gestorId, {
      ...stableFilters
    });
  }, [gestorId, stableFilters]);

  const loadPriceCompare = useCallback(async () => {
    if (mode !== 'gestor' || !gestorId) return null;
    return await analyticsService.getPriceComparisonBySegment({
      gestorId,
      periodo: normalizedPeriodo,
      ...stableFilters
    });
  }, [mode, gestorId, normalizedPeriodo, stableFilters]);

  // ✅ CONFIGURACIÓN DINÁMICA DE CHARTS SEGÚN MODO (MEMOIZADA)
  const chartsConfig = useMemo(() => {
    if (mode === 'direccion') {
      return [
        {
          key: 'ranking',
          title: `Ranking por ${normalizedMetric}`,
          subtitle: 'Top gestores por métrica seleccionada',
          icon: TrophyOutlined,
          color: theme.colors?.bmGreenPrimary || '#1890ff',
          span: { xs: 24, sm: 24, md: 12, lg: 12, xl: 12 },
          chartType: chartType || 'horizontal_bar'
        },
        {
          key: 'distribution', 
          title: 'Distribución por Centros',
          subtitle: 'Reparto de actividad por centros',
          icon: PieChartOutlined,
          color: theme.colors?.info || '#13c2c2',
          span: { xs: 24, sm: 24, md: 12, lg: 12, xl: 12 },
          chartType: chartType || 'donut'
        }
      ];
    } else {
      return [
        {
          key: 'ranking',
          title: 'Top Clientes',
          subtitle: 'Principales clientes por margen',
          icon: TrophyOutlined,
          color: theme.colors?.bmGreenPrimary || '#1890ff',
          span: { xs: 24, sm: 24, md: 12, lg: 8, xl: 8 },
          chartType: chartType || 'horizontal_bar'
        },
        {
          key: 'distribution',
          title: 'Mix de Productos',
          subtitle: 'Distribución de productos en cartera',
          icon: PieChartOutlined,
          color: theme.colors?.info || '#13c2c2',
          span: { xs: 24, sm: 24, md: 12, lg: 8, xl: 8 },
          chartType: chartType || 'donut'
        },
        {
          key: 'priceCompare',
          title: 'Precio: Real vs Estándar',
          subtitle: 'Comparativa con detección automática de segmento',
          icon: DollarCircleOutlined,
          color: theme.colors?.warning || '#faad14',
          span: { xs: 24, sm: 24, md: 24, lg: 8, xl: 8 },
          chartType: 'comparison',
          height: 420
        }
      ];
    }
  }, [mode, normalizedMetric, chartType]);

  // ✅ EFECTO PRINCIPAL OPTIMIZADO - CORREGIDO PARA CARGA INICIAL AUTOMÁTICA
  useEffect(() => {
    // ⭐ CAMBIO CLAVE: Solo verificar dependencias si ya se ha cargado antes
    if (lastDependencyKey.current !== null && lastDependencyKey.current === dependencyKey) {
      console.log('[InteractiveCharts] 🔄 Dependencies unchanged, skipping reload');
      return;
    }

    // Protección 2: Verificar parámetros mínimos requeridos
    if (!normalizedPeriodo) {
      console.log('[InteractiveCharts] ⚠️ No periodo provided');
      return;
    }

    if (mode === 'gestor' && !gestorId) {
      console.log('[InteractiveCharts] ⚠️ Gestor mode requires gestorId');
      return;
    }

    // ⭐ CAMBIO CLAVE: REMOVER EL FLAG GLOBAL DE LOADING QUE CAUSABA EL PROBLEMA
    // ❌ CÓDIGO REMOVIDO:
    // if (isLoadingRef.current) {
    //   console.log('[InteractiveCharts] ⚠️ Already loading, skipping...');
    //   return;
    // }
    // isLoadingRef.current = true;

    lastDependencyKey.current = dependencyKey;

    console.log('[InteractiveCharts] 🚀 Loading all charts with dependencies:', { 
      mode, 
      gestorId, 
      periodo: normalizedPeriodo,
      metric: normalizedMetric,
      dependencyKey: dependencyKey.substring(0, 100) + '...'
    });

    // Limpiar datos previos
    setChartsData({});
    setErrorStates({});

    // Función de carga principal MEJORADA
    const loadAllData = async () => {
      try {
        // ⭐ CAMBIO CLAVE: CARGAR EN PARALELO SIN PROTECCIÓN GLOBAL
        const loadPromises = [];

        // Ranking chart
        loadPromises.push(
          loadChartData(
            'ranking',
            mode === 'direccion' ? loadDirectionRanking : loadGestorRanking,
            true
          )
        );

        // Distribution chart
        loadPromises.push(
          loadChartData(
            'distribution',
            mode === 'direccion' ? loadDirectionDistribution : loadGestorDistribution,
            true
          )
        );

        // Price comparison (solo para gestor)
        if (mode === 'gestor' && gestorId) {
          loadPromises.push(
            loadChartData('priceCompare', loadPriceCompare, true)
          );
        }

        // Esperar todas las cargas
        await Promise.allSettled(loadPromises);

        console.log('[InteractiveCharts] ✅ All charts loaded successfully');

      } catch (error) {
        console.error('[InteractiveCharts] ❌ Error in loadAllData:', error);
      }
    };

    // Ejecutar carga
    loadAllData();

    // ✅ CLEANUP FUNCTION PARA CANCELAR REQUESTS AL DESMONTAR O CAMBIAR DEPENDENCIAS
    return () => {
      console.log('[InteractiveCharts] 🧹 Cleanup: Aborting all requests');
      
      Object.keys(abortControllers.current).forEach(key => {
        cleanupAbortController(key);
      });
    };

  }, [
    dependencyKey,
    mode,
    gestorId,
    normalizedPeriodo,
    normalizedMetric,
    loadChartData,
    loadDirectionRanking,
    loadDirectionDistribution,
    loadGestorRanking,
    loadGestorDistribution,
    loadPriceCompare,
    cleanupAbortController
  ]);

  // ✅ HANDLERS MEMOIZADOS
  const handleDataPointClick = useCallback((entity, chartKey) => {
    setSelectedEntity({ ...entity, chartKey });
    if (onSelectEntity) {
      onSelectEntity({ ...entity, chartKey, mode, periodo: normalizedPeriodo });
    }
  }, [onSelectEntity, mode, normalizedPeriodo]);

  const handleRefreshChart = useCallback((chartKey) => {
    console.log(`[InteractiveCharts] 🔄 Refreshing chart: ${chartKey}`);
    
    const loaderMap = {
      'ranking': mode === 'direccion' ? loadDirectionRanking : loadGestorRanking,
      'distribution': mode === 'direccion' ? loadDirectionDistribution : loadGestorDistribution,
      'priceCompare': loadPriceCompare
    };

    const loader = loaderMap[chartKey];
    if (loader) {
      loadChartData(chartKey, loader, true);
    }
    
    if (onReload) onReload();
  }, [
    mode, 
    loadDirectionRanking, 
    loadGestorRanking, 
    loadDirectionDistribution, 
    loadGestorDistribution, 
    loadPriceCompare, 
    loadChartData, 
    onReload
  ]);

  const handleRefreshAll = useCallback(() => {
    console.log('[InteractiveCharts] 🔄 Refreshing all charts');
    
    // Reset flags y limpiar cache
    lastDependencyKey.current = null;
    
    if (analyticsService.clearAnalyticsCache) {
      analyticsService.clearAnalyticsCache();
    }
    
    // Limpiar datos
    setChartsData({});
    setErrorStates({});
    
    if (onReload) onReload();
  }, [onReload]);

  // ✅ RENDERIZADOR DE CARDS MEMOIZADO
  const renderChartCard = useCallback((chartConfig) => {
    const data = chartsData[chartConfig.key];
    const loading = loadingStates[chartConfig.key];
    const error = errorStates[chartConfig.key];

    const getChartTitle = () => {
      if (chartConfig.key === 'priceCompare' && data?.meta?.segmentoNombre) {
        return `${chartConfig.title} - ${data.meta.segmentoNombre}`;
      }
      return chartConfig.title;
    };

    const getChartSubtitle = () => {
      if (chartConfig.key === 'priceCompare' && data?.meta) {
        const parts = [chartConfig.subtitle];
        if (data.meta.periodo) parts.push(data.meta.periodo);
        if (data.meta.mockData) parts.push('(Datos de demostración)');
        return parts.join(' - ');
      }
      return chartConfig.subtitle;
    };

    return (
      <Col key={chartConfig.key} {...chartConfig.span}>
        <Card
          title={
            <div style={{ display: 'flex', alignItems: 'center', gap: 12 }}>
              <chartConfig.icon style={{ 
                backgroundColor: `${chartConfig.color}15`, 
                color: chartConfig.color, 
                fontSize: 18, 
                padding: 8, 
                borderRadius: 6 
              }} />
              <div>
                <div style={{ 
                  fontWeight: 600, 
                  fontSize: 15, 
                  color: theme.colors?.textPrimary || '#333',
                  marginBottom: 2
                }}>
                  {getChartTitle()}
                  {loading && (
                    <Badge 
                      count="Cargando..." 
                      style={{ 
                        backgroundColor: theme.colors?.info || '#13c2c2',
                        marginLeft: 8,
                        fontSize: 10
                      }} 
                    />
                  )}
                </div>
                <Text type="secondary" style={{ fontSize: 12 }}>
                  {getChartSubtitle()}
                </Text>
              </div>
            </div>
          }
          extra={
            <Tooltip title="Actualizar datos">
              <Button
                size="small"
                icon={<ReloadOutlined />}
                loading={loading}
                onClick={() => handleRefreshChart(chartConfig.key)}
                style={{
                  borderColor: chartConfig.color,
                  color: chartConfig.color
                }}
              />
            </Tooltip>
          }
          style={{
            height: chartConfig.height || height,
            marginBottom: theme.spacing?.md || 16,
            borderRadius: theme.token?.borderRadius || 8,
            boxShadow: '0 4px 12px rgba(27, 94, 85, 0.12)',
            border: `1px solid ${theme.colors?.borderLight || '#e8e8e8'}`,
            transition: 'all 0.3s ease'
          }}
          styles={{
            body: {
              height: (chartConfig.height || height) - 75,
              padding: '16px 20px',
              overflow: 'hidden'
            }
          }}
        >
          {error && !loading && (
            <ErrorState
              error={error}
              onRetry={() => handleRefreshChart(chartConfig.key)}
              style={{ height: (chartConfig.height || height) - 90 }}
            />
          )}
          
          {!error && (
            <SimpleChart
              key={`${chartConfig.key}-${gestorId || 'default'}`}
              data={data}
              config={{ 
                chartType: chartConfig.chartType,
                topN: 10,
                showValues: true
              }}
              chartKey={chartConfig.key}
              height={(chartConfig.height || height) - 90}
              onDataClick={(entity) => handleDataPointClick(entity, chartConfig.key)}
            />
          )}
        </Card>
      </Col>
    );
  }, [chartsData, loadingStates, errorStates, height, handleRefreshChart, handleDataPointClick, gestorId]);

  return (
    <div className={`interactive-charts ${className}`} style={{ ...style, width: '100%' }}>
      {/* Header */}
      <div style={{ marginBottom: theme.spacing?.lg || 24 }}>
        <div style={{ 
          display: 'flex', 
          justifyContent: 'space-between', 
          alignItems: 'flex-start', 
          marginBottom: theme.spacing?.md || 16
        }}>
          <div>
            <Text style={{ 
              fontWeight: 600, 
              fontSize: 20, 
              color: theme.colors?.bmGreenPrimary || '#1890ff', 
              display: 'flex', 
              alignItems: 'center', 
              gap: 12 
            }}>
              <BarChartOutlined style={{ 
                padding: 10, 
                backgroundColor: `${theme.colors?.bmGreenPrimary || '#1890ff'}15`, 
                borderRadius: 8, 
                fontSize: 20 
              }} />
              Gráficos Interactivos
              <Badge 
                count={mode === 'gestor' ? 'Gestor' : 'Dirección'} 
                style={{ 
                  backgroundColor: theme.colors?.bmGreenPrimary || '#1890ff',
                  marginLeft: 4
                }} 
              />
            </Text>
            <div style={{ marginTop: 8 }}>
              <Text type="secondary" style={{ fontSize: 13 }}>
                {mode === 'direccion' 
                  ? `Dashboard corporativo • Métrica: ${normalizedMetric} • Período: ${normalizedPeriodo}`
                  : `Panel personalizado • Período: ${normalizedPeriodo}`}
                {mode === 'gestor' && gestorId && ` • Gestor: ${gestorId}`}
                {Object.keys(stableFilters).length > 0 && ' • Con filtros aplicados'}
              </Text>
            </div>
          </div>
          
          <Space>
            <Button 
              icon={<ReloadOutlined />} 
              onClick={handleRefreshAll}
              loading={Object.values(loadingStates).some(Boolean)}
              style={{ 
                borderColor: theme.colors?.bmGreenPrimary || '#1890ff',
                color: theme.colors?.bmGreenPrimary || '#1890ff'
              }}
            >
              Actualizar Todo
            </Button>
          </Space>
        </div>
      </div>

      {/* Grid principal */}
      <Row gutter={[20, 20]}>
        {chartsConfig.map(renderChartCard)}
      </Row>

      {/* Floating button */}
      <FloatButton
        icon={<ReloadOutlined />}
        type="primary"
        style={{
          right: 24,
          bottom: 24,
          backgroundColor: theme.colors?.bmGreenPrimary || '#1890ff'
        }}
        onClick={handleRefreshAll}
        tooltip={`Actualizar todos los gráficos - Modo: ${mode === 'direccion' ? 'Dirección' : 'Gestor'}`}
      />
    </div>
  );
};

InteractiveCharts.propTypes = {
  mode: PropTypes.oneOf(['direccion', 'gestor']).isRequired,
  periodo: PropTypes.oneOfType([PropTypes.string, PropTypes.object]).isRequired,
  gestorId: PropTypes.oneOfType([PropTypes.string, PropTypes.number]),
  metric: PropTypes.oneOf(['CONTRATOS', 'CLIENTES', 'MARGEN', 'INGRESOS', 'GASTOS']),
  chartType: PropTypes.oneOf(['horizontal_bar', 'bar', 'line', 'donut', 'stacked_bar']),
  filters: PropTypes.object,
  height: PropTypes.number,
  onReload: PropTypes.func,
  onSelectEntity: PropTypes.func,
  externalChartConfig: PropTypes.object,
  onChartConfigChange: PropTypes.func,
  className: PropTypes.string,
  style: PropTypes.object
};

export default React.memo(InteractiveCharts);
