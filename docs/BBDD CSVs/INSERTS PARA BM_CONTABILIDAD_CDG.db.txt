-- 3. Recrear tabla con CONTRATO_ID opcional
CREATE TABLE MOVIMIENTOS_CONTRATOS (
    MOVIMIENTO_ID INTEGER PRIMARY KEY AUTOINCREMENT,
    EMPRESA_ID INTEGER NOT NULL,
    FECHA DATE NOT NULL,
    CONTRATO_ID INTEGER,  -- ← SIN NOT NULL para permitir gastos operativos
    CENTRO_CONTABLE INTEGER NOT NULL,
    CUENTA_ID TEXT NOT NULL,
    DIVISA TEXT NOT NULL,
    IMPORTE REAL NOT NULL,
    LINEA_CUENTA_RESULTADOS TEXT NOT NULL,
    CONCEPTO_GESTION TEXT NOT NULL,
    FOREIGN KEY (EMPRESA_ID) REFERENCES MAESTRO_EMPRESAS(EMPRESA_ID),
    FOREIGN KEY (CONTRATO_ID) REFERENCES MAESTRO_CONTRATOS(CONTRATO_ID),
    FOREIGN KEY (CENTRO_CONTABLE) REFERENCES MAESTRO_CENTROS(CENTRO_ID),
    FOREIGN KEY (CUENTA_ID) REFERENCES MAESTRO_CUENTAS(CUENTA_ID),
    FOREIGN KEY (LINEA_CUENTA_RESULTADOS) REFERENCES MAESTRO_LINEA_CDR(COD_LINEA_CDR)
);

);
